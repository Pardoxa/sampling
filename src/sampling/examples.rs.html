3/// # Example Coin flips
4///
5/// Lets assume we do n Coinflips and want to measure the probability for the number of times,
6/// this results in Head. This means, the number of times the coin flip returned head is the `energy`
7///
8/// Of course, for this example there is an analytic solution.
9///
10/// For the implementation of the coin flip sequence and Markov chain of it, please look in the [source code](../../../src/sampling/examples/coin_flips.rs.html)
11///
12/// Now A detailed example for large deviation simulations with **comparison to analytical results**
13///
14/// The files created in this example can be found below
15/// ```
16///
17/// use rand::SeedableRng;
18/// use rand_pcg::Pcg64;
19/// use sampling::{*, examples::coin_flips::*};
20/// use std::fs::File;
21/// use std::io::{BufWriter, Write};
22/// use statrs::distribution::{Binomial, Discrete};
23/// use std::num::NonZeroUsize;
24///
25/// // length of coin flip sequence
26/// let n = 20;
27/// let interval_count = NonZeroUsize::new(3).unwrap();
28///
29/// // create histogram. The result of our `energy` (number of heads) can be anything between 0 and n
30/// let hist = HistUsizeFast::new_inclusive(0, n).unwrap();
31///
32/// // now the overlapping histograms for sampling
33/// // lets create 3 histograms. The parameter Overlap should be larger than 0. Normally, 1 is sufficient
34/// let hist_list = hist.overlapping_partition(interval_count, 1).unwrap();
35/// // alternatively you could also create the histograms in the desired interval.
36/// // Just make sure, that they overlap
37///
38/// // create rng to seed all other rngs
39/// let mut rng = Pcg64::seed_from_u64(834628956578);
40///
41/// // now create ensembles (could be combined with wl creation)
42/// // note: You could also create one ensemble and clone it instead of creating different ones
43/// let ensembles: Vec<_> = (0..interval_count.get()).map(|_| {
44///     CoinFlipSequence::new(
45///         n,
46///         Pcg64::from_rng(&mut rng)
47///     )
48/// }).collect();
49///
50/// // Now the Wang Landau simulation. First create the struct
51/// // (here as Vector, since we want to use 3 overlapping intervals)
52/// let mut wl_list: Vec<_> = ensembles.into_iter()
53///     .zip(hist_list.into_iter())
54///     .map(|(ensemble, histogram)| {
55///         WangLandau1T::new(
56///             0.000001, // arbitrary threshold for `log_f`(see paper),
57///                      // you have to try what is good for your model
58///             ensemble,
59///             Pcg64::from_rng(&mut rng),
60///             1,  // stepsize 1 is sufficient for this problem
61///             histogram,
62///             100 // every 100 steps: check if WL can refine factor f
63///         ).unwrap()
64///     }).collect();
65///
66/// // Now we have to initialize the wl with a valid state
67/// // as the simulation has to start in the interval one wants to measure.
68/// // Since the energy landscape is quite simple, here a greedy approach is good enough.
69///
70/// wl_list.iter_mut()
71///     .for_each(|wl|{
72///         wl.init_greedy_heuristic(
73///             |coin_seq| Some(coin_seq.head_count()),
74///             Some(10_000) // if no valid state is found after 10_000
75///                          // this returns an Err. If you do not want a step limit,
76///                          // you can use None here
77///         ).expect("Unable to find valid state within 10_000 steps!");
78///     });
79///
80/// // Now our ensemble is initialized. Time for the Wang Landau Simulation.
81/// // You can do that in different ways.
82/// // I will show this by doing it differently for our three intervals
83///
84/// // First, the simplest one. Just simulate until the threshold for `log_f` is reached
85/// wl_list[0].wang_landau_convergence(
86///     |coin_seq| Some(coin_seq.head_count())
87/// );
88///
89/// // Secondly, I only have a limited amount of time.
90/// // Lets say, I have 1 minute at most.
91/// let start_time = std::time::Instant::now();
92/// wl_list[1].wang_landau_while(
93///     |coin_seq| Some(coin_seq.head_count()),
94///     |_| start_time.elapsed().as_secs() <= 60
95/// );
96///
97/// // Or lets say, I want to limit the number of steps to 100_000
98/// wl_list[2].wang_landau_while(
99///     |coin_seq| Some(coin_seq.head_count()),
100///     |state| state.step_counter() <= 100_000
101/// );
102///
103/// // Now, lets see if our last two simulations did indeed finish:
104/// // This one did
105/// assert!(wl_list[1].is_finished());
106/// // This simulation did not finish
107/// assert!(!wl_list[2].is_finished());
108///
109/// // If a simulation did not finish, you could, e.g., store the state (`wl_list[2]`) using serde.
110/// // Then you could continue the simulation later on.
111/// // I recommend the crate `bincode` for storing
112///
113/// // lets resume the simulation for now
114/// wl_list[2].wang_landau_convergence(
115///     |coin_seq| Some(coin_seq.head_count())
116/// );
117/// // it finished
118/// assert!(wl_list[2].is_finished());
119///
120/// // Since our simulations did all finish, lets see what our distribution looks like
121/// // Lets glue them together. We use our original histogram for that.
122/// let mut glue_job =
123///     GlueJob::new_from_slice(&wl_list, LogBase::Base10);
124/// let mut glued = glue_job
125///     .average_merged_and_aligned()
126///     .expect("Unable to glue results. Look at error message");
127///
128/// // now, lets print our result
129/// glued.set_stat_write_verbosity(GlueWriteVerbosity::AccumulatedStats);
130/// glued.write(std::io::stdout()).unwrap();
131///
132/// // or store it into a file
133/// let file = File::create("coin_flip_log_density.dat").unwrap();
134/// let buf = BufWriter::new(file);
135/// glued.write(buf).unwrap();
136///
137/// // now, lets check if our results are actually any good.
138/// // lets compare that to the analytical result
139///
140/// // Since the library I am going to use lets me directly calculate the natural
141/// // logaritm of the probability, I now switch the base of our glue results
142/// glued.switch_base();
143/// let ln_prob = glued.glued();
144///
145/// // Then create the `true` results:
146/// let binomial = Binomial::new(0.5, n as u64).unwrap();
147///
148/// let ln_prob_true: Vec<_> = (0..=n)
149///     .map(|k| binomial.ln_pmf(k as u64))
150///     .collect();
151///
152/// // lets write that in a file, so we can use gnuplot to plot the result
153/// let comp_file = File::create("coin_flip_compare.dat").unwrap();
154/// let mut buf = BufWriter::new(comp_file);
155///
156/// // lets also calculate the maximum difference between the two solutions
157/// let mut max_ln_dif = std::f64::NEG_INFINITY;
158/// let mut max_dif = std::f64::NEG_INFINITY;
159///
160/// writeln!(buf, "#head_count Numeric_ln_prob Analytic_ln_prob ln_dif dif").unwrap();
161/// for (index, (numeric, analytic)) in ln_prob.iter().zip(ln_prob_true.iter()).enumerate()
162/// {
163///     let ln_dif = numeric - analytic;
164///     max_ln_dif = ln_dif.abs().max(max_ln_dif);
165///     let dif = numeric.exp() - analytic.exp();
166///     max_dif = dif.abs().max(max_dif);
167///     writeln!(buf, "{} {:e} {:e} {:e} {:e}", index, numeric, analytic, ln_dif, dif).unwrap();
168/// }
169///
170/// println!("Max_ln_dif = {}", max_ln_dif);
171/// println!("Max_dif = {}", max_dif);
172///
173/// // in this case, the max difference of the natural logarithms
174/// // of the probabilities is smaller than 0.03
175/// assert!(max_ln_dif < 0.03);
176/// // and the max absolute difference is smaller than 0.0009
177/// assert!(max_dif < 0.0009);
178///
179/// // But we can do better. Lets refine the results with entropic sampling
180/// // first, convert the wl simulations in entropic sampling simulations
181/// let mut entropic_list: Vec<_> = wl_list
182///     .into_iter()
183///     .map(|wl| EntropicSampling::from_wl(wl).unwrap())
184///     .collect();
185///
186///
187/// // Now, while doing that, lets also create a heatmap.
188/// // Lets say, we want to see, how the number of times `Head` occurred in the sequence
189/// // correlates to the maximum number of `Heads` in a row in that sequence.
190///
191/// // In this case, the heatmap is symmetric and we already have a histogram of correct sice
192/// let mut heatmap = HeatmapU::new(
193///     hist.clone(),
194///     hist.clone()
195/// );
196///
197/// entropic_list.iter_mut()
198///     .for_each(|entr|{
199///         entr.entropic_sampling(
200///             |coin_seq| Some(coin_seq.head_count()),
201///             |state| {
202///                 let head_count = *state.energy();
203///                 let heads_in_row = state.ensemble().max_heads_in_a_row();
204///                 heatmap.count(head_count, heads_in_row)
205///                     .expect("Value outside heatmap?");
206///             }
207///         )
208///     });
209///
210/// // Now, lets see our refined results:
211/// let mut glue_job =
212///     GlueJob::new_from_slice(&entropic_list, LogBase::Base10);
213/// // Note that the library also allows you to mix REWL, REES, Entropic and Wanglandau simulations
214/// // with the GlueJob, if you deem it necessary. We won't do that here though
215///
216/// // after creating the GlueJob we need to do the merging:
217/// let mut glued = glue_job
218///     .average_merged_and_aligned()
219///     .expect("Unable to glue results. Look at error message");
220/// // to also write Statistics of the simulation as comments
221/// glued.set_stat_write_verbosity(GlueWriteVerbosity::IntervalStatsAndAccumulatedStats);
222///
223/// // lets store our result
224/// let file = File::create("coin_flip_log_density_entropic.dat").unwrap();
225/// let buf = BufWriter::new(file);
226/// glued.write(buf).unwrap();
227///
228/// // now, lets compare with the analytical results again
229/// // Again, calculate to base e
230/// // (Note that we could have calculated to base e directly by choosing so in the GlueJob)
231/// glued.switch_base();
232/// let ln_prob = glued.glued();
233///
234///
235/// // lets write that in a file, so we can use gnuplot to plot the result
236/// let comp_file = File::create("coin_flip_compare_entr.dat").unwrap();
237/// let mut buf = BufWriter::new(comp_file);
238///
239/// // lets also calculate the maximum difference between the two solutions
240/// let mut max_ln_dif = std::f64::NEG_INFINITY;
241/// let mut max_dif = std::f64::NEG_INFINITY;
242///
243/// writeln!(buf, "#head_count Numeric_ln_prob Analytic_ln_prob ln_dif dif").unwrap();
244/// for (index, (numeric, analytic)) in ln_prob.iter().zip(ln_prob_true.iter()).enumerate()
245/// {
246///     let ln_dif = numeric - analytic;
247///     max_ln_dif = ln_dif.abs().max(max_ln_dif);
248///     let dif = numeric.exp() - analytic.exp();
249///     max_dif = dif.abs().max(max_dif);
250///     writeln!(buf, "{} {:e} {:e} {:e} {:e}", index, numeric, analytic, ln_dif, dif).unwrap();
251/// }
252///
253/// println!("Max_ln_dif = {}", max_ln_dif);
254/// println!("Max_dif = {}", max_dif);
255///
256/// // in this case, the max difference of the natural logarithms
257/// //of the probabilities is smaller than 0.026
258/// assert!(max_ln_dif < 0.026);
259/// // and the max absolut difference is smaller than 0.0007
260/// assert!(max_dif < 0.0007);
261///
262/// // That would be the final result for our probability
263/// // density than. As you can see, it is very very
264/// // close to the analytical result.
265///
266/// // Now, lets see, how our heatmap looks:
267/// let mut settings = GnuplotSettings::new();
268/// settings.x_label("#Heads")
269///     .y_label("Max heads in row")
270///     .terminal(GnuplotTerminal::PDF("heatmap_coin_flips".to_owned()));
271///
272/// // lets normalize coloumwise
273/// // This way, the scale of our heatmap tells us the conditional probability
274/// // P(Number of heads in a rom | number of heads) of how many heads in a row were
275/// // part of that sequence given the total number of heads that occurred in the sequence
276/// let heatmap = heatmap.heatmap_normalized_columns();
277///
278/// // now create gnuplot file
279/// let file = File::create("coin_heatmap.gp").unwrap();
280/// let buf = BufWriter::new(file);
281/// heatmap.gnuplot(
282///     buf,
283///     settings
284/// ).unwrap();
285///
286/// // now you can use gnuplot to plot the heatmap
