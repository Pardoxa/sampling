25/// Includes statistics about the intervals from which the gluing result stems from
26pub struct GlueStats{
27    /// List of the interval simulation stats of the intervals used to create the glue result
28    pub interval_stats: Vec<IntervalSimStats>,
29    /// List of the round trips of the intervals used to create the glue result
30    pub roundtrips: Vec<usize>
31}
32
33impl GlueStats{
34    /// # Write the Glued Stats in a human readable way
35    /// * This is the verbose form
36    /// * every line this writes will be starting with '#', to mark it as comment for 
37    ///   gnuplot and co.
38    pub fn write_verbose<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()>
39    {
40        if !self.interval_stats.is_empty()
41        {
42            writeln!(writer, "#Interval Stats")?;
43
44            for (index, interval) in self.interval_stats.iter().enumerate()
45            {
46                writeln!(writer, "#Stats for Interval {index}")?;
47                interval.write(&mut writer)?;
48                writeln!(writer, "#")?;
49            }
50        }
51
52        if !self.roundtrips.is_empty(){
53            let mut min_roundtrips = usize::MAX;
54            write!(writer, "#Roundtrips (higher is better):")?;
55            for &r in self.roundtrips.iter()
56            {
57                min_roundtrips = min_roundtrips.min(r);
58                write!(writer, " {r}")?;
59            }
60            writeln!(writer)?;
61            writeln!(writer, "#Minimum of performed Roundtrips {min_roundtrips}")?;
62        }
63        Ok(())
64    }
65
66    /// Writes stats to a file
67    /// 
68    /// These stats contain key data on various stuff, like the factor f or the total number of steps, the global rejection rate
69    /// and the minimum number of performed roundtrips. 
70    /// 
71    /// Information that is currently unavailable might be skipped.
72    /// 
73    /// All outputted lines have a preceding '#', to mark them as comments 
74    pub fn write_accumulated<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()>{
75        if !self.interval_stats.is_empty(){
76            let acc = AccumulatedIntervalStats::generate_stats(&self.interval_stats);
77            acc.write(&mut writer)?;
78        }
79        if !self.roundtrips.is_empty(){
80            let min_roundtrips = self.roundtrips.iter().min().unwrap();
81            writeln!(writer, "#Minimum of performed Roundtrips {min_roundtrips}")?;
82        }
83        Ok(())
84    }
85}

242    /// # Write the Glued in a human readable format
243    /// * You probably want to use this ;)
244    pub fn write<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()>
245    {
246        write!(writer, "#")?;
247        self.encapsulating_histogram.write_header(&mut writer)?;
248        write!(writer, " log_merged")?;
249        
250        for i in 0..self.aligned.len()
251        {
252            write!(writer, " interval_{i}")?;
253        }
254        writeln!(writer)?;
255
256        writeln!(writer, "#log: {:?}", self.base)?;
257        self.write_stats(&mut writer)?;
258
259        let mut alinment_helper = self.alignment_helper();
260
261        for (&log_prob, bin) in self.glued
262            .iter()
263            .zip(self.encapsulating_histogram.display_bin_iter())
264        {
265            H::write_bin(&bin, &mut writer)?;
266            write!(writer, " {:e}", log_prob)?;
267            for (i, counter) in alinment_helper.iter_mut().enumerate()
268            {
269                if *counter < 0 {
270                    write!(writer, " NaN")?
271                } else {
272                    let val = self.aligned[i].get(*counter as usize);
273                    match val {
274                        Some(&v) => write!(writer, " {:e}", v)?,
275                        None => write!(writer, " NaN")?
276                    }
277                }
278                *counter += 1;
279            }
280            writeln!(writer)?;
281        }
282        Ok(())
283    }
284
285    fn alignment_helper(&self) -> Vec<isize>
286    {
287        let mut alinment_helper: Vec<_> = std::iter::once(0)
288            .chain(
289                self.alignment.iter()
290                    .map(|&v| -(v as isize))
291            ).collect();
292
293        let mut sum = 0;
294        alinment_helper.iter_mut()
295            .for_each(
296                |val| 
297                {
298                    let old = sum;
299                    sum += *val;
300                    *val += old;
301                }
302            );
303        alinment_helper
304    }

306    fn write_stats<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()>
307    {
308        if let Some(stats) = self.stats.as_ref()
309        {
310            match self.write_verbosity
311            {
312                GlueWriteVerbosity::NoStats => {
313                    Ok(())
314                },
315                GlueWriteVerbosity::AccumulatedStats => {
316                    stats.write_accumulated(writer)
317                },
318                GlueWriteVerbosity::IntervalStats => {
319                    stats.write_verbose(writer)
320                },
321                GlueWriteVerbosity::IntervalStatsAndAccumulatedStats => {
322                    stats.write_verbose(&mut writer)?;
323                    stats.write_accumulated(writer)
324                }
325            }
326        } else {
327            Ok(())
328        }
329        
330    }

332    /// # Write the normalized probability density function
333    /// The function will be normalized by using the binsize 
334    /// you specify (uniform binsize is assumed).
335    /// 
336    /// Then it will write the merged log probability as well as the intervals 
337    /// into the writer you specified
338    pub fn write_rescaled<W: std::io::Write>(
339        &self,
340        mut writer: W,
341        bin_size: f64,
342        starting_point: f64
343    ) -> std::io::Result<()>
344    {
345        write!(writer, "#bin log_merged")?;
346        for i in 0..self.aligned.len()
347        {
348            write!(writer, " interval_{i}")?;
349        }
350        writeln!(writer)?;
351        writeln!(writer, "#log: {:?}", self.base)?;
352        self.write_stats(&mut writer)?;
353        let bin_size_recip = bin_size.recip();
354
355        let rescale = match self.base {
356            LogBase::BaseE => bin_size_recip.ln(),
357            LogBase::Base10 => bin_size_recip.log10(),
358        };
359
360        let mut alinment_helper = self.alignment_helper();
361
362        for (index, log_prob) in self.glued
363            .iter()
364            .map(|s| *s + rescale)
365            .enumerate()
366        {
367            let bin = starting_point + index as f64 * bin_size;
368            write!(writer, "{} {:e}", bin, log_prob)?;
369            for (i, counter) in alinment_helper.iter_mut().enumerate()
370            {
371                if *counter < 0 {
372                    write!(writer, " NaN")?
373                } else {
374                    let val = self.aligned[i].get(*counter as usize);
375                    match val {
376                        Some(&v) => 
377                        {
378                            let rescaled = v + rescale;
379                            write!(writer, " {:e}", rescaled)?
380                        },
381                        None => write!(writer, " NaN")?
382                    }
383                }
384                *counter += 1;
385            }
386            writeln!(writer)?;
387        }
388        Ok(())
389    }

393pub(crate) fn calc_merge_points(alignment: &[usize], derivatives: &[Vec<f64>]) -> Vec<usize>
394{
395    derivatives.iter()
396        .zip(derivatives[1..].iter())
397        .zip(alignment.iter())
398        .map(
399            |((left, right), &align)|
400            {
401                (align..)
402                    .zip(
403                        left[align..].iter()
404                        .zip(right.iter())
405                    )
406                    .map(
407                        |(index, (&left, &right))|
408                        {
409                            (index, (left - right).abs())
410                        }
411                    ).fold( (usize::MAX, f64::INFINITY),
412                        |a, b|
413                        if a.1 < b.1 {
414                            a
415                        } else {
416                            b
417                        }
418                    ).0
419            }
420        ).collect()
421}

423/// # Calculate the probability density function from overlapping intervals
424/// `log_prob` is a vector of the logarithmic non-normalized probability densities
425/// 
426/// `hists` is a vector of the corresponding histograms
427/// 
428/// `LogBase`: Which base do the logarithmic probabilities have?
429/// 
430/// This uses a derivative merge, that works similar to: [derivative_merged_log_prob_and_aligned](crate::rees::ReplicaExchangeEntropicSampling::derivative_merged_log_prob_and_aligned)
431/// 
432/// The [Glued] allows you to easily write the probability density function to a file
433pub fn derivative_merged_and_aligned<H, Hist, T>(
434    mut log_prob: Vec<Vec<f64>>,
435    hists: &[H],
436    log_base: LogBase
437) -> Result<Glued<Hist, T>, HistErrors>
438where Hist: HistogramCombine + Histogram,
439    H: Borrow<Hist>
440{
441
442    // get the log_probabilities - the walkers over the same intervals are merged
443    log_prob
444        .iter_mut()
445        .for_each(
446            |v| 
447            {
448                subtract_max(v);
449            }
450        );
451    // get the derivative, for merging later
452    let derivatives: Vec<_> = log_prob.iter()
453        .map(|v| derivative_merged(v))
454        .collect();
455    let e_hist = Hist::encapsulating_hist(hists)?;
456    let alignment  = hists.iter()
457        .zip(hists.iter().skip(1))
458        .map(|(left, right)| left.borrow().align(right.borrow()))
459        .collect::<Result<Vec<_>, _>>()?;
460    
461    
462    let merge_points = calc_merge_points(&alignment, &derivatives);
463
464    // Not even one Interval - this has to be an error
465    if log_prob.is_empty() {
466        return Err(HistErrors::EmptySlice)
467    }
468
469    let mut merged_log_prob = vec![f64::NAN; e_hist.bin_count()];
470
471    
472    // Nothing to align, only one interval here
473    if alignment.is_empty() {
474        log_base.norm_log_prob(&mut log_prob[0]);
475        
476        let merged_prob = log_prob[0].clone();
477        let r = 
478            Glued::new_unchecked(
479                e_hist,
480                merged_prob,
481                log_prob,
482                log_base,
483                alignment,
484                None
485            );
486        return Ok(r);
487    }
488
489    merged_log_prob[..=merge_points[0]]
490        .copy_from_slice(&log_prob[0][..=merge_points[0]]);
491
492
493    // https://play.rust-lang.org/?version=stable&mode=debug&edition=2018&gist=2dcb7b7a3be78397d34657ece42aa851
494    let mut align_sum = 0;
495
496    for ((&a, &mp), log_prob_p1) in alignment.iter()
497        .zip(merge_points.iter())
498        .zip(log_prob.iter_mut().skip(1))
499    {
500        let position_l = mp + align_sum;
501        align_sum += a;
502        let left = mp - a;
503
504        let shift = merged_log_prob[position_l] - log_prob_p1[left];
505
506        // shift
507        log_prob_p1
508            .iter_mut()
509            .for_each(|v| *v += shift);
510
511        merged_log_prob[position_l..]
512            .iter_mut()
513            .zip(log_prob_p1[left..].iter())
514            .for_each(
515                |(merge, &val)|
516                {
517                    *merge = val;
518                }
519            );
520
521
522    }
523
524    let shift = log_base.norm_log_prob(&mut merged_log_prob);
525
526    log_prob.iter_mut()
527        .for_each(
528            |interval|
529            interval.iter_mut()
530                .for_each(|val| *val -= shift)
531        );
532
533    let glued = 
534        Glued::new_unchecked(
535            e_hist, 
536            merged_log_prob, 
537            log_prob, 
538            LogBase::BaseE, 
539            alignment,
540            None
541        );
542        
543    Ok(
544        glued
545    )
546}

548pub(crate) fn norm_ln_prob(ln_prob: &mut[f64]) -> f64
549{
550    let max = subtract_max(ln_prob);
551    // calculate actual sum in non log space
552    let sum = ln_prob.iter()
553        .fold(0.0, |acc, &val| {
554            if val.is_finite(){
555               acc +  val.exp()
556            } else {
557                acc
558            }
559        }
560    );
561
562    let shift = sum.ln();
563
564    ln_prob.iter_mut()
565        .for_each(|val| *val -= shift);
566
567    shift - max
568
569}

571pub(crate) fn norm_log10_prob(log10_prob: &mut[f64]) -> f64
572{
573    let max = subtract_max(log10_prob);
574    // calculate actual sum in non log space
575    let sum = log10_prob.iter()
576        .fold(0.0, |acc, &val| {
577            if val.is_finite(){
578               acc +  10.0_f64.powf(val)
579            } else {
580                acc
581            }
582        }
583    );
584
585    let shift = sum.log10();
586
587    log10_prob.iter_mut()
588        .for_each(|val| *val -= shift);
589
590    shift - max
591
592}

594/// # Calculate the probability density function from overlapping intervals
595/// `log_prob` is a vector of the logarithmic non-normalized probability densities
596/// 
597/// `hists` is a vector of the corresponding histograms
598/// 
599/// `LogBase`: Which base do the logarithmic probabilities have?
600/// 
601/// This uses a average merge, which first align all intervals and then merges 
602/// the probability densities by averaging in the logarithmic space
603/// 
604/// The [Glued] allows you to easily write the probability density function to a file
605pub fn average_merged_and_aligned<Hist, H, T>(
606    mut log_prob: Vec<Vec<f64>>,
607    hists: &[H],
608    log_base: LogBase
609) -> Result<Glued<Hist, T>, HistErrors>
610where Hist: HistogramCombine + Histogram,
611    H: Borrow<Hist>
612{
613    // get the log_probabilities - the walkers over the same intervals are merged
614    log_prob
615        .iter_mut()
616        .for_each(
617            |v| 
618            {
619                subtract_max(v);
620            }
621        );
622    let e_hist = Hist::encapsulating_hist(hists)?;
623    let alignment  = hists
624        .iter()
625        .zip(hists.iter().skip(1))
626        .map(|(left, right)| left.borrow().align(right.borrow()))
627        .collect::<Result<Vec<_>, _>>()?;
628
629    if alignment.is_empty(){
630        // entering this means we only have 1 interval!
631        assert_eq!(log_prob.len(), 1);
632        log_base.norm_log_prob(&mut log_prob[0]);
633        
634        let glued = log_prob[0].clone();
635        return Ok(
636                Glued{
637                base: LogBase::Base10,
638                encapsulating_histogram: e_hist,
639                aligned: log_prob,
640                glued,
641                alignment,
642                marker: PhantomData,
643                stats: None,
644                write_verbosity: GlueWriteVerbosity::NoStats
645            }
646        );
647    }
648
649    // calc z
650    let z_vec = calc_z(&log_prob, &alignment)
651        .expect("Unable to calculate Z in glueing");
652
653    // correct height
654    height_correction(&mut log_prob, &z_vec);
655    // renaming
656    let mut aligned_intervals = log_prob;
657
658    // glueing together
659    let mut glued_log_density = glue_no_derive(e_hist.bin_count(), &aligned_intervals, &alignment)
660        .expect("Glue error!");
661
662    // now norm the result
663    let shift = log_base.norm_log_prob(&mut glued_log_density);
664
665    aligned_intervals
666        .iter_mut()
667        .flat_map(|vec| vec.iter_mut())
668        .for_each(|v| *v -= shift);
669
670    Ok(
671        Glued{
672            base: log_base,
673            encapsulating_histogram: e_hist,
674            aligned: aligned_intervals,
675            glued: glued_log_density,
676            alignment,
677            marker: PhantomData,
678            stats: None,
679            write_verbosity: GlueWriteVerbosity::NoStats
680        }
681    )
682}

685fn glue_no_derive(size: usize, log10_vec: &[Vec<f64>], alignment: &[usize]) -> Result<Vec<f64>, GlueErrors>
686{
687    let mut glue_log_density = vec![f64::NAN; size];
688
689
690    // init - first interval can be copied for better performance
691    let first_log = match log10_vec.first(){
692        Some(interval) => interval.as_slice(),
693        None => return Err(GlueErrors::EmptyList)
694    };
695   
696    glue_log_density[0..first_log.len()].copy_from_slice(first_log);
697    let mut glue_count = vec![0_usize; glue_log_density.len()];
698    
699    #[allow(clippy::needless_range_loop)]
700    for (index, val) in first_log.iter().enumerate() {
701        if val.is_finite(){
702            glue_count[index] = 1;
703        }
704    }
705
706    let mut offset = 0;
707    for (i, log_vec) in log10_vec.iter().enumerate().skip(1)
708    {
709        offset += alignment[i-1];
710
711        glue_log_density.iter_mut()
712            .zip(glue_count.iter_mut())
713            .skip(offset)
714            .zip(log_vec.iter())
715            .for_each(
716                |((glued, count), &prob)|
717                {
718                    if prob.is_finite(){
719                        *count += 1;
720                        *glued = if glued.is_finite(){
721                            *glued + prob
722                        } else {
723                            prob
724                        };
725                    }
726                    
727                }
728            );
729    }
730
731    glue_log_density.iter_mut()
732        .zip(glue_count.iter())
733        .for_each(|(log, &count)| {
734            if count > 0 {
735                *log /= count as f64;
736            } else {
737                *log = f64::NAN;
738            }
739        });
740    
741    Ok(glue_log_density)
742}

745fn calc_z(log10_vec: &[Vec<f64>], alignment: &[usize]) -> Result<Vec<f64>, GlueErrors>
746{
747    let mut z_vec = Vec::with_capacity(alignment.len());
748    for (i, &align) in alignment.iter().enumerate()
749    {
750        let prob_right = &log10_vec[i+1];
751        let prob_left = &log10_vec[i][align..];
752        
753        let mut counter: usize = 0;
754        let mut sum = 0.0;
755        for (p, c) in prob_left.iter().zip(prob_right.iter())
756        {
757            if p.is_finite() && c.is_finite(){
758                counter += 1;
759                sum += p - c;
760            }
761        }
762        
763        let mut z = sum / counter as f64;
764        // also correct for adjustment of prev
765        if let Some(val) = z_vec.last() {
766            z += val;
767        }
768        z_vec.push(z);
769    }
