19/// Includes statistics about the intervals from which the gluing result stems from
20pub struct GlueStats {
21    /// List of the interval simulation stats of the intervals used to create the glue result
22    pub interval_stats: Vec<IntervalSimStats>,
23    /// List of the round trips of the intervals used to create the glue result
24    pub roundtrips: Vec<usize>,
25}
26
27impl GlueStats {
28    /// # Write the Glued Stats in a human readable way
29    /// * This is the verbose form
30    /// * every line this writes will be starting with '#', to mark it as comment for
31    ///   gnuplot and co.
32    pub fn write_verbose<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()> {
33        if !self.interval_stats.is_empty() {
34            writeln!(writer, "#Interval Stats")?;
35
36            for (index, interval) in self.interval_stats.iter().enumerate() {
37                writeln!(writer, "#Stats for Interval {index}")?;
38                interval.write(&mut writer)?;
39                writeln!(writer, "#")?;
40            }
41        }
42
43        if !self.roundtrips.is_empty() {
44            let mut min_roundtrips = usize::MAX;
45            write!(writer, "#Roundtrips (higher is better):")?;
46            for &r in self.roundtrips.iter() {
47                min_roundtrips = min_roundtrips.min(r);
48                write!(writer, " {r}")?;
49            }
50            writeln!(writer)?;
51            writeln!(writer, "#Minimum of performed Roundtrips {min_roundtrips}")?;
52        }
53        Ok(())
54    }
55
56    /// Writes stats to a file
57    ///
58    /// These stats contain key data on various stuff, like the factor f or the total number of steps, the global rejection rate
59    /// and the minimum number of performed roundtrips.
60    ///
61    /// Information that is currently unavailable might be skipped.
62    ///
63    /// All outputted lines have a preceding '#', to mark them as comments
64    pub fn write_accumulated<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()> {
65        if !self.interval_stats.is_empty() {
66            let acc = AccumulatedIntervalStats::generate_stats(&self.interval_stats);
67            acc.write(&mut writer)?;
68        }
69        if !self.roundtrips.is_empty() {
70            let min_roundtrips = self.roundtrips.iter().min().unwrap();
71            writeln!(writer, "#Minimum of performed Roundtrips {min_roundtrips}")?;
72        }
73        Ok(())
74    }
75}

205                self.aligned
206                    .iter_mut()
207                    .for_each(|interval| ln_to_log10(interval));
208                self.base = LogBase::Base10;
209            }

219    /// # Write the Glued in a human readable format
220    /// * You probably want to use this ;)
221    pub fn write<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()> {
222        write!(writer, "#")?;
223        self.encapsulating_histogram.write_header(&mut writer)?;
224        write!(writer, " log_merged")?;
225
226        for i in 0..self.aligned.len() {
227            write!(writer, " interval_{i}")?;
228        }
229        writeln!(writer)?;
230
231        writeln!(writer, "#log: {:?}", self.base)?;
232        self.write_stats(&mut writer)?;
233
234        let mut alinment_helper = self.alignment_helper();
235
236        for (&log_prob, bin) in self
237            .glued
238            .iter()
239            .zip(self.encapsulating_histogram.display_bin_iter())
240        {
241            H::write_bin(&bin, &mut writer)?;
242            write!(writer, " {:e}", log_prob)?;
243            for (i, counter) in alinment_helper.iter_mut().enumerate() {
244                if *counter < 0 {
245                    write!(writer, " NaN")?
246                } else {
247                    let val = self.aligned[i].get(*counter as usize);
248                    match val {
249                        Some(&v) => write!(writer, " {:e}", v)?,
250                        None => write!(writer, " NaN")?,
251                    }
252                }
253                *counter += 1;
254            }
255            writeln!(writer)?;
256        }
257        Ok(())
258    }
259
260    fn alignment_helper(&self) -> Vec<isize> {
261        let mut alinment_helper: Vec<_> = std::iter::once(0)
262            .chain(self.alignment.iter().map(|&v| -(v as isize)))
263            .collect();
264
265        let mut sum = 0;
266        alinment_helper.iter_mut().for_each(|val| {
267            let old = sum;
268            sum += *val;
269            *val += old;
270        });
271        alinment_helper
272    }
273
274    fn write_stats<W: std::io::Write>(&self, mut writer: W) -> std::io::Result<()> {
275        if let Some(stats) = self.stats.as_ref() {
276            match self.write_verbosity {
277                GlueWriteVerbosity::NoStats => Ok(()),
278                GlueWriteVerbosity::AccumulatedStats => stats.write_accumulated(writer),
279                GlueWriteVerbosity::IntervalStats => stats.write_verbose(writer),
280                GlueWriteVerbosity::IntervalStatsAndAccumulatedStats => {
281                    stats.write_verbose(&mut writer)?;
282                    stats.write_accumulated(writer)
283                }
284            }
285        } else {
286            Ok(())
287        }
288    }

290    /// # Write the normalized probability density function
291    /// The function will be normalized by using the binsize
292    /// you specify (uniform binsize is assumed).
293    ///
294    /// Then it will write the merged log probability as well as the intervals
295    /// into the writer you specified
296    pub fn write_rescaled<W: std::io::Write>(
297        &self,
298        mut writer: W,
299        bin_size: f64,
300        starting_point: f64,
301    ) -> std::io::Result<()> {
302        write!(writer, "#bin log_merged")?;
303        for i in 0..self.aligned.len() {
304            write!(writer, " interval_{i}")?;
305        }
306        writeln!(writer)?;
307        writeln!(writer, "#log: {:?}", self.base)?;
308        self.write_stats(&mut writer)?;
309        let bin_size_recip = bin_size.recip();
310
311        let rescale = match self.base {
312            LogBase::BaseE => bin_size_recip.ln(),
313            LogBase::Base10 => bin_size_recip.log10(),
314        };
315
316        let mut alinment_helper = self.alignment_helper();
317
318        for (index, log_prob) in self.glued.iter().map(|s| *s + rescale).enumerate() {
319            let bin = starting_point + index as f64 * bin_size;
320            write!(writer, "{} {:e}", bin, log_prob)?;
321            for (i, counter) in alinment_helper.iter_mut().enumerate() {
322                if *counter < 0 {
323                    write!(writer, " NaN")?
324                } else {
325                    let val = self.aligned[i].get(*counter as usize);
326                    match val {
327                        Some(&v) => {
328                            let rescaled = v + rescale;
329                            write!(writer, " {:e}", rescaled)?
330                        }
331                        None => write!(writer, " NaN")?,
332                    }
333                }
334                *counter += 1;
335            }
336            writeln!(writer)?;
337        }
338        Ok(())
339    }

342pub(crate) fn calc_merge_points(alignment: &[usize], derivatives: &[Vec<f64>]) -> Vec<usize> {
343    derivatives
344        .iter()
345        .zip(derivatives[1..].iter())
346        .zip(alignment.iter())
347        .map(|((left, right), &align)| {
348            (align..)
349                .zip(left[align..].iter().zip(right.iter()))
350                .map(|(index, (&left, &right))| (index, (left - right).abs()))
351                .fold(
352                    (usize::MAX, f64::INFINITY),
353                    |a, b| if a.1 < b.1 { a } else { b },
354                )
355                .0
356        })
357        .collect()
358}

360/// # Calculate the probability density function from overlapping intervals
361/// `log_prob` is a vector of the logarithmic non-normalized probability densities
362///
363/// `hists` is a vector of the corresponding histograms
364///
365/// `LogBase`: Which base do the logarithmic probabilities have?
366///
367/// This uses a derivative merge, that works similar to: [derivative_merged_log_prob_and_aligned](crate::rees::ReplicaExchangeEntropicSampling::derivative_merged_log_prob_and_aligned)
368///
369/// The [Glued] allows you to easily write the probability density function to a file
370pub fn derivative_merged_and_aligned<H, Hist, T>(
371    mut log_prob: Vec<Vec<f64>>,
372    hists: &[H],
373    log_base: LogBase,
374) -> Result<Glued<Hist, T>, HistErrors>
375where
376    Hist: HistogramCombine + Histogram,
377    H: Borrow<Hist>,
378{
379    // get the log_probabilities - the walkers over the same intervals are merged
380    log_prob.iter_mut().for_each(|v| {
381        subtract_max(v);
382    });
383    // get the derivative, for merging later
384    let derivatives: Vec<_> = log_prob.iter().map(|v| derivative_merged(v)).collect();
385    let e_hist = Hist::encapsulating_hist(hists)?;
386    let alignment = hists
387        .iter()
388        .zip(hists.iter().skip(1))
389        .map(|(left, right)| left.borrow().align(right.borrow()))
390        .collect::<Result<Vec<_>, _>>()?;
391
392    let merge_points = calc_merge_points(&alignment, &derivatives);
393
394    // Not even one Interval - this has to be an error
395    if log_prob.is_empty() {
396        return Err(HistErrors::EmptySlice);
397    }
398
399    let mut merged_log_prob = vec![f64::NAN; e_hist.bin_count()];
400
401    // Nothing to align, only one interval here
402    if alignment.is_empty() {
403        log_base.norm_log_prob(&mut log_prob[0]);
404
405        let merged_prob = log_prob[0].clone();
406        let r = Glued::new_unchecked(e_hist, merged_prob, log_prob, log_base, alignment, None);
407        return Ok(r);
408    }
409
410    merged_log_prob[..=merge_points[0]].copy_from_slice(&log_prob[0][..=merge_points[0]]);
411
412    // https://play.rust-lang.org/?version=stable&mode=debug&edition=2018&gist=2dcb7b7a3be78397d34657ece42aa851
413    let mut align_sum = 0;
414
415    for ((&a, &mp), log_prob_p1) in alignment
416        .iter()
417        .zip(merge_points.iter())
418        .zip(log_prob.iter_mut().skip(1))
419    {
420        let position_l = mp + align_sum;
421        align_sum += a;
422        let left = mp - a;
423
424        let shift = merged_log_prob[position_l] - log_prob_p1[left];
425
426        // shift
427        log_prob_p1.iter_mut().for_each(|v| *v += shift);
428
429        merged_log_prob[position_l..]
430            .iter_mut()
431            .zip(log_prob_p1[left..].iter())
432            .for_each(|(merge, &val)| {
433                *merge = val;
434            });
435    }
436
437    let shift = log_base.norm_log_prob(&mut merged_log_prob);
438
439    log_prob
440        .iter_mut()
441        .for_each(|interval| interval.iter_mut().for_each(|val| *val -= shift));
442
443    let glued = Glued::new_unchecked(
444        e_hist,
445        merged_log_prob,
446        log_prob,
447        LogBase::BaseE,
448        alignment,
449        None,
450    );
451
452    Ok(glued)
453}

455pub(crate) fn norm_ln_prob(ln_prob: &mut [f64]) -> f64 {
456    let max = subtract_max(ln_prob);
457    // calculate actual sum in non log space
458    let sum = ln_prob.iter().fold(0.0, |acc, &val| {
459        if val.is_finite() {
460            acc + val.exp()
461        } else {
462            acc
463        }
464    });
465
466    let shift = sum.ln();
467
468    ln_prob.iter_mut().for_each(|val| *val -= shift);
469
470    shift - max
471}

473pub(crate) fn norm_log10_prob(log10_prob: &mut [f64]) -> f64 {
474    let max = subtract_max(log10_prob);
475    // calculate actual sum in non log space
476    let sum = log10_prob.iter().fold(0.0, |acc, &val| {
477        if val.is_finite() {
478            acc + 10.0_f64.powf(val)
479        } else {
480            acc
481        }
482    });
483
484    let shift = sum.log10();
485
486    log10_prob.iter_mut().for_each(|val| *val -= shift);
487
488    shift - max
489}

491/// # Calculate the probability density function from overlapping intervals
492/// `log_prob` is a vector of the logarithmic non-normalized probability densities
493///
494/// `hists` is a vector of the corresponding histograms
495///
496/// `LogBase`: Which base do the logarithmic probabilities have?
497///
498/// This uses a average merge, which first align all intervals and then merges
499/// the probability densities by averaging in the logarithmic space
500///
501/// The [Glued] allows you to easily write the probability density function to a file
502pub fn average_merged_and_aligned<Hist, H, T>(
503    mut log_prob: Vec<Vec<f64>>,
504    hists: &[H],
505    log_base: LogBase,
506) -> Result<Glued<Hist, T>, HistErrors>
507where
508    Hist: HistogramCombine + Histogram,
509    H: Borrow<Hist>,
510{
511    // get the log_probabilities - the walkers over the same intervals are merged
512    log_prob.iter_mut().for_each(|v| {
513        subtract_max(v);
514    });
515    let e_hist = Hist::encapsulating_hist(hists)?;
516    let alignment = hists
517        .iter()
518        .zip(hists.iter().skip(1))
519        .map(|(left, right)| left.borrow().align(right.borrow()))
520        .collect::<Result<Vec<_>, _>>()?;
521
522    if alignment.is_empty() {
523        // entering this means we only have 1 interval!
524        assert_eq!(log_prob.len(), 1);
525        log_base.norm_log_prob(&mut log_prob[0]);
526
527        let glued = log_prob[0].clone();
528        return Ok(Glued {
529            base: LogBase::Base10,
530            encapsulating_histogram: e_hist,
531            aligned: log_prob,
532            glued,
533            alignment,
534            marker: PhantomData,
535            stats: None,
536            write_verbosity: GlueWriteVerbosity::NoStats,
537        });
538    }
539
540    // calc z
541    let z_vec = calc_z(&log_prob, &alignment).expect("Unable to calculate Z in glueing");
542
543    // correct height
544    height_correction(&mut log_prob, &z_vec);
545    // renaming
546    let mut aligned_intervals = log_prob;
547
548    // glueing together
549    let mut glued_log_density =
550        glue_no_derive(e_hist.bin_count(), &aligned_intervals, &alignment).expect("Glue error!");
551
552    // now norm the result
553    let shift = log_base.norm_log_prob(&mut glued_log_density);
554
555    aligned_intervals
556        .iter_mut()
557        .flat_map(|vec| vec.iter_mut())
558        .for_each(|v| *v -= shift);
559
560    Ok(Glued {
561        base: log_base,
562        encapsulating_histogram: e_hist,
563        aligned: aligned_intervals,
564        glued: glued_log_density,
565        alignment,
566        marker: PhantomData,
567        stats: None,
568        write_verbosity: GlueWriteVerbosity::NoStats,
569    })
570}

572fn glue_no_derive(
573    size: usize,
574    log10_vec: &[Vec<f64>],
575    alignment: &[usize],
576) -> Result<Vec<f64>, GlueErrors> {
577    let mut glue_log_density = vec![f64::NAN; size];
578
579    // init - first interval can be copied for better performance
580    let first_log = match log10_vec.first() {
581        Some(interval) => interval.as_slice(),
582        None => return Err(GlueErrors::EmptyList),
583    };
584
585    glue_log_density[0..first_log.len()].copy_from_slice(first_log);
586    let mut glue_count = vec![0_usize; glue_log_density.len()];
587
588    #[allow(clippy::needless_range_loop)]
589    for (index, val) in first_log.iter().enumerate() {
590        if val.is_finite() {
591            glue_count[index] = 1;
592        }
593    }
594
595    let mut offset = 0;
596    for (i, log_vec) in log10_vec.iter().enumerate().skip(1) {
597        offset += alignment[i - 1];
598
599        glue_log_density
600            .iter_mut()
601            .zip(glue_count.iter_mut())
602            .skip(offset)
603            .zip(log_vec.iter())
604            .for_each(|((glued, count), &prob)| {
605                if prob.is_finite() {
606                    *count += 1;
607                    *glued = if glued.is_finite() {
608                        *glued + prob
609                    } else {
610                        prob
611                    };
612                }
613            });
614    }
615
616    glue_log_density
617        .iter_mut()
618        .zip(glue_count.iter())
619        .for_each(|(log, &count)| {
620            if count > 0 {
621                *log /= count as f64;
622            } else {
623                *log = f64::NAN;
624            }
625        });
626
627    Ok(glue_log_density)
628}

630fn calc_z(log10_vec: &[Vec<f64>], alignment: &[usize]) -> Result<Vec<f64>, GlueErrors> {
631    let mut z_vec = Vec::with_capacity(alignment.len());
632    for (i, &align) in alignment.iter().enumerate() {
633        let prob_right = &log10_vec[i + 1];
634        let prob_left = &log10_vec[i][align..];
635
636        let mut counter: usize = 0;
637        let mut sum = 0.0;
638        for (p, c) in prob_left.iter().zip(prob_right.iter()) {
639            if p.is_finite() && c.is_finite() {
640                counter += 1;
641                sum += p - c;
642            }
643        }
644
645        let mut z = sum / counter as f64;
646        // also correct for adjustment of prev
647        if let Some(val) = z_vec.last() {
648            z += val;
649        }
650        z_vec.push(z);
651    }
652    Ok(z_vec)
