7/// Possible errors that can occur during gluing together
8/// WangLandau intervals or Entropic Sampling intervals
9pub enum GlueErrors {
10    /// `original_hist.borders_clone()` failed
11    BorderCreation(HistErrors),
12    /// Nothing to be glued, glue interval list was empty
13    EmptyList,
14    /// Binary search failed - PartialOrd::partial_cmp returned None
15    BinarySearch,
16    /// # Glue interval and intervals to be glued do not match
17    /// * Likely `original_hist` is to small
18    OutOfBounds,
19    /// The intervals need to overlap, otherwise no gluing can occur
20    NoOverlap,
21}
22
23impl From<HistErrors> for GlueErrors {
24    fn from(e: HistErrors) -> Self {
25        GlueErrors::BorderCreation(e)
26    }
27}

29/// # Normalize log10 probability density
30/// * input: Slice containing log10 of (non normalized) probability density
31/// * afterwards, it will be normalized, i.e., sum_i 10^log10_density\[i\] â‰ˆ 1
32pub fn norm_log10_sum_to_1(log10_density: &mut [f64]) {
33    // prevent errors due to small or very large numbers
34    subtract_max(log10_density);
35
36    // calculate actual sum in non log space
37    let sum = log10_density.iter().fold(0.0, |acc, &val| {
38        if val.is_finite() {
39            acc + 10_f64.powf(val)
40        } else {
41            acc
42        }
43    });
44
45    let sum = sum.log10();
46    log10_density.iter_mut().for_each(|val| *val -= sum);
47}

49pub(crate) fn height_correction(log10_vec: &mut [Vec<f64>], z_vec: &[f64]) {
50    log10_vec
51        .iter_mut()
52        .skip(1)
53        .zip(z_vec.iter())
54        .for_each(|(vec, &z)| {
55            vec.iter_mut().for_each(|val| {
56                *val += z;
57            })
58        });
59}

61/// subtracts maximum, if it is finite
62pub(crate) fn subtract_max(list: &mut [f64]) -> f64 {
63    let max = list.iter().copied().fold(f64::NAN, f64::max);
64
65    if max.is_finite() {
66        list.iter_mut().for_each(|val| *val -= max);
67    }
68    max
69}
