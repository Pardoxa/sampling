12/// Error states, that entropic sampling, or the creation of `EntropicSamplingAdaptive`
13/// could encounter
14pub enum EntropicErrors {
15    /// # source (`WangLandauAdaptive`) was in an invalid state
16    /// * did you forget to use one of the `init*` methods to initialize a valid
17    ///   WangLandau state?
18    InvalidWangLandau,
19
20    /// Still in the process of gathering statistics
21    /// Not enough to make an estimate
22    NotEnoughStatistics,
23
24    /// Still Gathering Statistics, this is only an estimate!
25    EstimatedStatistic(Vec<f64>),
26
27    /// Invalid trial step. Is your max_step smaller than your min_step?
28    InvalidMinMaxTrialSteps,
29
30    /// # Possible reasons
31    /// * `log_density.len()` and `histogram.bin_count()` are not equal
32    /// * not all values of `log_density` are finite
33    InvalidLogDensity,
34
35    /// You are trying to have a `min_best_of_count` that is
36    /// larger than the total steps you try!
37    InvalidBestof,
38}

73impl<Hist, R, E, S, Res, Energy> GlueAble<Hist>
74    for EntropicSamplingAdaptive<Hist, R, E, S, Res, Energy>
75where
76    Hist: Clone + Histogram,
77{
78    fn push_glue_entry_ignoring(&self, job: &mut GlueJob<Hist>, ignore_idx: &[usize]) {
79        if !ignore_idx.contains(&0) {
80            let mut missing_steps = 0;
81            if self.step_count() < self.step_goal() {
82                missing_steps = (self.step_goal() - self.step_count()) as u64;
83            }
84            let rejected = self.total_entr_steps_rejected() as u64;
85            let accepted = self.total_entr_steps_accepted() as u64;
86
87            let stats = IntervalSimStats {
88                sim_progress: SimProgress::MissingSteps(missing_steps),
89                interval_sim_type: SimulationType::Entropic,
90                rejected_steps: rejected,
91                accepted_steps: accepted,
92                replica_exchanges: None,
93                proposed_replica_exchanges: None,
94                merged_over_walkers: NonZeroUsize::new(1).unwrap(),
95            };
96
97            let glue_entry = GlueEntry {
98                hist: self.hist().clone(),
99                prob: self.log_density_refined(),
100                log_base: LogBase::BaseE,
101                interval_stats: stats,
102            };
103            job.collection.push(glue_entry);
104        }
105    }
106}

108impl<Hist, R, E, S, Res, T> TryFrom<WangLandauAdaptive<Hist, R, E, S, Res, T>>
109    for EntropicSamplingAdaptive<Hist, R, E, S, Res, T>
110where
111    Hist: Histogram,
112    R: Rng,
113{
114    type Error = EntropicErrors;
115    fn try_from(mut wl: WangLandauAdaptive<Hist, R, E, S, Res, T>) -> Result<Self, Self::Error> {
116        if wl.old_bin.is_none() || wl.old_energy.is_none() {
117            return Err(EntropicErrors::InvalidWangLandau);
118        }
119        let wl_steps_rejected = wl.total_steps_rejected();
120        let wl_steps_accepted = wl.total_steps_accepted();
121        wl.accepted_step_hist.iter_mut().for_each(|v| *v = 0);
122        wl.rejected_step_hist.iter_mut().for_each(|v| *v = 0);
123        wl.best_of_steps.clear();
124        wl.histogram.reset();
125        wl.trial_list.shuffle(&mut wl.rng);
126
127        Ok(Self {
128            wl_steps_rejected,
129            wl_steps_accepted,
130            counter: 0,
131            steps: wl.steps,
132            step_res_marker: wl.step_res_marker,
133            log_density: wl.log_density,
134            old_energy: wl.old_energy.unwrap(),
135            old_bin: wl.old_bin.unwrap(),
136            accepted_step_hist: wl.accepted_step_hist,
137            rejected_step_hist: wl.rejected_step_hist,
138            total_steps_accepted: 0,
139            total_steps_rejected: 0,
140            min_step: wl.min_step,
141            min_best_of_count: wl.min_best_of_count,
142            best_of_steps: wl.best_of_steps,
143            best_of_threshold: wl.best_of_threshold,
144            rng: wl.rng,
145            trial_list: wl.trial_list,
146            ensemble: wl.ensemble,
147            step_count: 0,
148            step_goal: wl.step_count,
149            histogram: wl.histogram,
150            adjust_bestof_every: 10usize.max(4 * wl.check_refine_every),
151        })
152    }
153}

155impl<Hist, R, E, S, Res, T> EntropicSamplingAdaptive<Hist, R, E, S, Res, T> {
156    /// # Current state of the Ensemble
157    #[inline]
158    pub fn ensemble(&self) -> &E {
159        &self.ensemble
160    }
161
162    /// # Energy of ensemble
163    /// * assuming `energy_fn` (see `self.entropic_step` etc.)
164    ///   is deterministic and will always give the same result for the same ensemble,
165    ///   this returns the energy of the current ensemble
166    #[inline]
167    pub fn energy(&self) -> &T {
168        &self.old_energy
169    }
170
171    /// # Number of entropic steps done until now
172    /// * will be reset by [`self.refine_estimate`](#method.refine_estimate)
173    #[inline]
174    pub fn step_count(&self) -> usize {
175        self.step_count
176    }
177
178    /// # Number of entropic steps to be performed
179    /// * if `self` was created from `WangLandauAdaptive`,
180    ///   `step_goal` will be equal to the number of WangLandau steps, that were performed
181    #[inline]
182    pub fn step_goal(&self) -> usize {
183        self.step_goal
184    }
185
186    /// # Number of entropic steps to be performed
187    /// * set the number of steps to be performed by entropic sampling
188    #[inline]
189    pub fn set_step_goal(&mut self, step_goal: usize) {
190        self.step_goal = step_goal;
191    }
192
193    /// # Smallest possible markov step (`m_steps` of MarkovChain trait) by entropic step
194    #[inline]
195    pub fn min_step_size(&self) -> usize {
196        self.min_step
197    }
198
199    /// # Largest possible markov step (`m_steps` of MarkovChain trait) by entropic step
200    #[inline]
201    pub fn max_step_size(&self) -> usize {
202        self.min_step + self.accepted_step_hist.len() - 1
203    }
204
205    /// # Currently used best of
206    /// * might have length 0, if statistics are still being gathered
207    /// * otherwise this contains the step sizes, from which the next step size
208    ///   is drawn uniformly
209    #[inline]
210    pub fn best_of_steps(&self) -> &Vec<usize> {
211        &self.best_of_steps
212    }
213
214    /// # Fraction of steps accepted since the statistics were reset the last time
215    /// * (steps accepted since last reset) / (steps since last reset)
216    /// * `NaN` if no steps were performed yet
217    pub fn fraction_accepted_current(&self) -> f64 {
218        let accepted: usize = self.accepted_step_hist.iter().sum();
219        let total = accepted + self.rejected_step_hist.iter().sum::<usize>();
220        if total == 0 {
221            f64::NAN
222        } else {
223            accepted as f64 / total as f64
224        }
225    }
226
227    /// # total number of entropic steps, that were accepted
228    pub fn total_entr_steps_accepted(&self) -> usize {
229        self.total_steps_accepted + self.accepted_step_hist.iter().sum::<usize>()
230    }
231
232    /// # total number of entropic steps, that were rejected
233    pub fn total_entr_steps_rejected(&self) -> usize {
234        self.total_steps_rejected + self.rejected_step_hist.iter().sum::<usize>()
235    }
236
237    /// # Fraction of steps accepted since the creation of `self`
238    /// * `NaN` if no steps were performed yet
239    pub fn fraction_accepted_total_entropic(&self) -> f64 {
240        let total_acc = self.total_entr_steps_accepted();
241        let total_steps = total_acc + self.total_entr_steps_rejected();
242
243        if total_steps == 0 {
244            f64::NAN
245        } else {
246            total_acc as f64 / total_steps as f64
247        }
248    }
249
250    /// * returns the (non normalized) log_density estimate log(P(E)), with which the simulation was started
251    /// * if you created this from a WangLandau simulation, this is the result of the WangLandau Simulation
252    pub fn log_density_estimate(&self) -> &Vec<f64> {
253        &self.log_density
254    }
255
256    /// calculates the (non normalized) log_density estimate log(P(E)) according to the [paper](#entropic-sampling-made-easy)
257    pub fn log_density_refined(&self) -> Vec<f64>
258    where
259        Hist: Histogram,
260    {
261        let mut log_density = Vec::with_capacity(self.log_density.len());
262        log_density.extend(
263            self.log_density
264                .iter()
265                .zip(self.histogram.hist().iter())
266                .map(|(&log_p, &h)| {
267                    if h == 0 {
268                        log_p
269                    } else {
270                        log_p + (h as f64).ln()
271                    }
272                }),
273        );
274        log_density
275    }
276
277    /// # Return current state of histogram
278    pub fn hist(&self) -> &Hist {
279        &self.histogram
280    }

282impl<Hist, R, E, S, Res, T> EntropicSamplingAdaptive<Hist, R, E, S, Res, T>
283where
284    Hist: Histogram,
285    R: Rng,
286{
287    /// # Creates EntropicSamplingAdaptive from a `WangLandauAdaptive` state
288    /// * `WangLandauAdaptive` state needs to be valid, i.e., you must have called one of the `init*` methods
289    /// - this ensures, that the members `old_energy` and `old_bin` are not `None`
290    pub fn from_wl_adaptive(
291        wl: WangLandauAdaptive<Hist, R, E, S, Res, T>,
292    ) -> Result<Self, EntropicErrors> {
293        wl.try_into()
294    }
295
296    /// # Calculates `self.log_density_refined` and uses that as estimate for a the entropic sampling simulation
297    /// * returns old estimate
298    /// # prepares `self` for a new entropic simulation
299    /// * sets new estimate for log(P(E))
300    /// * resets statistic gathering
301    /// * resets step_count
302    pub fn refine_estimate(&mut self) -> Vec<f64> {
303        let mut estimate = self.log_density_refined();
304        std::mem::swap(&mut estimate, &mut self.log_density);
305        self.counter = 0;
306        self.step_count = 0;
307        self.best_of_steps.clear();
308        self.histogram.reset();
309        self.trial_list.shuffle(&mut self.rng);
310
311        self.total_steps_accepted += self.accepted_step_hist.iter().sum::<usize>();
312        self.accepted_step_hist
313            .iter_mut()
314            .for_each(|entry| *entry = 0);
315
316        self.total_steps_rejected += self.rejected_step_hist.iter().sum::<usize>();
317        self.rejected_step_hist
318            .iter_mut()
319            .for_each(|entry| *entry = 0);
320
321        estimate
322    }
323
324    /// # How often to adjust `bestof_steps`?
325    /// * if you try to set a value smaller 10, it will be set to 10
326    /// * will re-evaluate the statistics every `adjust_bestof_every` steps,
327    /// - this will not start new statistics gathering but just trigger a reevaluation of
328    ///   the gathered statistics (should be O(max_stepsize - min_stepsize))
329    #[inline]
330    pub fn set_adjust_bestof_every(&mut self, adjust_bestof_every: usize) {
331        self.adjust_bestof_every = adjust_bestof_every.max(10);
332    }
333
334    /// Is the simulation in the process of rebuilding the statistics,
335    /// i.e., is it currently trying many different step sizes?
336    #[inline]
337    pub fn is_rebuilding_statistics(&self) -> bool {
338        self.counter < self.trial_list.len()
339    }
340
341    fn statistic_bin_not_hit(&self) -> bool {
342        self.accepted_step_hist
343            .iter()
344            .zip(self.rejected_step_hist.iter())
345            .any(|(a, r)| a + r == 0)
346    }
347
348    /// # Estimate accept/reject statistics
349    /// * contains list of estimated probabilities for accepting a step of corresponding step size
350    /// * list\[i\] corresponds to step size `i + self.min_step`
351    /// * O(trial_step_max - trial_step_min)
352    pub fn estimate_statistics(&self) -> Result<Vec<f64>, WangLandauErrors> {
353        let calc_estimate = || {
354            let mut estimate = Vec::with_capacity(self.accepted_step_hist.len());
355            estimate.extend(
356                self.accepted_step_hist
357                    .iter()
358                    .zip(self.rejected_step_hist.iter())
359                    .map(|(&a, &r)| a as f64 / (a + r) as f64),
360            );
361            estimate
362        };
363        if self.is_rebuilding_statistics() {
364            if self.statistic_bin_not_hit() {
365                Err(WangLandauErrors::NotEnoughStatistics)
366            } else {
367                Err(WangLandauErrors::EstimatedStatistic(calc_estimate()))
368            }
369        } else {
370            Ok(calc_estimate())
371        }
372    }
373
374    fn generate_bestof(&mut self) {
375        let statistics = self.estimate_statistics().unwrap();
376        let mut heap = BinaryHeap::with_capacity(statistics.len());
377        heap.extend(
378            statistics
379                .into_iter()
380                .enumerate()
381                .map(|(index, prob)| ProbIndex::new(prob, index)),
382        );
383        while let Some(p_i) = heap.pop() {
384            if p_i.is_best_of(self.best_of_threshold)
385                || self.best_of_steps.len() < self.min_best_of_count
386            {
387                let step_size = p_i.index + self.min_step;
388                self.best_of_steps.push(step_size);
389            } else {
390                break;
391            }
392        }
393    }
394
395    fn adjust_bestof(&mut self) {
396        self.best_of_steps.clear();
397        self.generate_bestof();
398    }
399
400    fn get_stepsize(&mut self) -> usize {
401        match self.trial_list.get(self.counter) {
402            None => {
403                if self.best_of_steps.is_empty() {
404                    self.generate_bestof();
405                } else if self.counter % self.adjust_bestof_every == 0 {
406                    self.adjust_bestof();
407                }
408                *self.best_of_steps.choose(&mut self.rng).unwrap()
409            }
410            Some(&step_size) => step_size,
411        }
412    }
413
414    #[inline]
415    fn count_accepted(&mut self, size: usize) {
416        self.accepted_step_hist[size - self.min_step] += 1;
417        self.counter += 1;
418    }
419
420    #[inline]
421    fn count_rejected(&mut self, size: usize) {
422        self.rejected_step_hist[size - self.min_step] += 1;
423        self.counter += 1;
424    }
425
426    /// **panics** if index is invalid
427    #[inline(always)]
428    fn metropolis_acception_prob(&self, new_bin: usize) -> f64 {
429        (self.log_density[self.old_bin] - self.log_density[new_bin]).exp()
430    }
431}

440    /// # Entropic sampling
441    /// * performs `self.entropic_step(energy_fn)` until `condition` is false
442    /// * **Note**: you have access to the current step_count (`self.step_count()`)
443    /// # Parameter
444    /// * `energy_fn` function calculating `Some(energy)` of the system
445    ///   or rather the Parameter of which you wish to obtain the probability distribution.
446    ///   If there are any states, for which the calculation is invalid, `None` should be returned
447    /// * steps resulting in ensembles for which `energy_fn(&mut ensemble)` is `None`
448    ///   will always be rejected
449    /// * **Important** `energy_fn`: should be the same as used for Wang Landau, otherwise the results will be wrong!
450    /// * `print_fn`: see below
451    /// # Correlations
452    /// * if you want to measure correlations between "energy" and other measurable quantities,
453    ///   use `print_fn`, which will be called after each step - use this function to write to
454    ///   a file or whatever you desire
455    /// * Note: You do not have to recalculate the energy, if you need it in `print_fn`:
456    ///   just call `self.energy()`
457    /// * you have access to your ensemble with `self.ensemble()`
458    /// * if you do not need it, you can use `|_|{}` as `print_fn`
459    ///  ## Safety
460    /// * While you do have mutable access to the ensemble, the energy function should not change the
461    ///   ensemble in a way, which affects the next calculation of the energy
462    /// * This is intended for usecases, where the energy calculation is more efficient with mutable access, e.g., through using a
463    ///   buffer stored in the ensemble
464    /// * Note: I chose to make this function unsafe to force users to acknowledge the (purely logical) limitations
465    ///   regarding the usage of the mutable ensemble. From a programming point of view this will not lead to
466    ///   any undefined behavior or such regardless of if the user fulfills the requirements
467    pub unsafe fn entropic_sampling_while_unsafe<F, G, W>(
468        &mut self,
469        mut energy_fn: F,
470        mut print_fn: G,
471        mut condition: W,
472    ) where
473        F: FnMut(&mut E) -> Option<T>,
474        G: FnMut(&Self),
475        W: FnMut(&Self) -> bool,
476    {
477        while condition(self) {
478            self.entropic_step_unsafe(&mut energy_fn);
479            print_fn(self);
480        }
481    }

582    where
583        F: FnMut(&mut E) -> Option<T>,
584        G: FnMut(&mut Self),
585    {
586        while self.step_count < self.step_goal {
587            self.entropic_step_unsafe(&mut energy_fn);
588            print_fn(self);
589        }
590    }
591
592    /// # Entropic sampling
593    /// * performs `self.entropic_step(energy_fn)` until `self.step_count == self.step_goal`
594    /// # Parameter
595    /// * `energy_fn` function calculating `Some(energy)` of the system
596    ///   or rather the Parameter of which you wish to obtain the probability distribution.
597    ///   If there are any states, for which the calculation is invalid, `None` should be returned
598    /// * steps resulting in ensembles for which `energy_fn(&mut ensemble)` is `None`
599    ///   will always be rejected
600    /// * **Important** `energy_fn`: should be the same as used for Wang Landau, otherwise the results will be wrong!
601    /// * `print_fn`: see below
602    /// # Correlations
603    /// * if you want to measure correlations between "energy" and other measurable quantities,
604    ///   use `print_fn`, which will be called after each step - use this function to write to
605    ///   a file or whatever you desire
606    /// * Note: You do not have to recalculate the energy, if you need it in `print_fn`:
607    ///   just call `self.energy()`
608    /// * you have access to your ensemble with `self.ensemble()`
609    /// * if you do not need it, you can use `|_|{}` as `print_fn`
610    pub fn entropic_sampling<F, G>(&mut self, mut energy_fn: F, mut print_fn: G)
611    where
612        F: FnMut(&E) -> Option<T>,
613        G: FnMut(&Self),
614    {
615        while self.step_count < self.step_goal {
616            self.entropic_step(&mut energy_fn);
617            print_fn(self);
618        }
619    }
620
621    /// # Entropic sampling using an accumulating markov step
622    /// * performs `self.entropic_step_acc(&mut energy_fn)` until `self.step_count == self.step_goal`
623    /// # Parameter
624    /// * `energy_fn` function calculating the energy `E` of the system
625    ///   (or rather the Parameter of which you wish to obtain the probability distribution)
626    ///   during the markov steps, which can be more efficient.
627    /// * **Important** `energy_fn`: should be the same as used for Wang Landau, otherwise the results will be wrong!
628    /// * `print_fn`: see below
629    /// # Correlations
630    /// * if you want to measure correlations between "energy" and other measurable quantities,
631    ///   use `print_fn`, which will be called after each step - use this function to write to
632    ///   a file or whatever you desire
633    /// * Note: You do not have to recalculate the energy, if you need it in `print_fn`:
634    ///   just call `self.energy()`
635    /// * you have access to your ensemble with `self.ensemble()`
636    /// * if you do not need it, you can use `|_|{}` as `print_fn`
637    pub fn entropic_sampling_acc<F, G>(&mut self, mut energy_fn: F, mut print_fn: G)
638    where
639        F: FnMut(&E, &S, &mut T),
640        G: FnMut(&Self),
641    {
642        while self.step_count < self.step_goal {
643            self.entropic_step_acc(&mut energy_fn);
644            print_fn(self);
645        }
646    }
647
648    /// # Entropic step
649    /// * performs a single step
650    /// # Parameter
651    /// * `energy_fn` function calculating `Some(energy)` of the system
652    ///   or rather the Parameter of which you wish to obtain the probability distribution.
653    ///   If there are any states, for which the calculation is invalid, `None` should be returned
654    /// * steps resulting in ensembles for which `energy_fn(&mut ensemble)` is `None`
655    ///   will always be rejected
656    /// # Important
657    /// * `energy_fn`: should be the same as used for Wang Landau, otherwise the results will be wrong!
658    ///  ## Safety
659    /// * While you do have mutable access to the ensemble, the energy function should not change the
660    ///   ensemble in a way, which affects the next calculation of the energy
661    /// * This is intended for usecases, where the energy calculation is more efficient with mutable access, e.g., through using a
662    ///   buffer stored in the ensemble
663    /// * Note: I chose to make this function unsafe to force users to acknowledge the (purely logical) limitations
664    ///   regarding the usage of the mutable ensemble. From a programming point of view this will not lead to
665    ///   any undefined behavior or such regardless of if the user fulfills the requirements
666    pub unsafe fn entropic_step_unsafe<F>(&mut self, mut energy_fn: F)
667    where
668        F: FnMut(&mut E) -> Option<T>,
669    {
670        self.step_count += 1;
671        let step_size = self.get_stepsize();
672
673        self.ensemble.m_steps(step_size, &mut self.steps);
674
675        let current_energy = match energy_fn(&mut self.ensemble) {
676            Some(energy) => energy,
677            None => {
678                self.ensemble.steps_rejected(&self.steps);
679                self.count_rejected(step_size);
680                self.histogram.increment_index(self.old_bin).unwrap();
681                self.ensemble.undo_steps_quiet(&self.steps);
682                return;
683            }
684        };
685
686        self.entropic_step_helper(current_energy);
687    }
688
689    /// # Entropic step
690    /// * performs a single step
691    /// # Parameter
692    /// * `energy_fn` function calculating `Some(energy)` of the system
693    ///   or rather the Parameter of which you wish to obtain the probability distribution.
694    ///   If there are any states, for which the calculation is invalid, `None` should be returned
695    /// * steps resulting in ensembles for which `energy_fn(&mut ensemble)` is `None`
696    ///   will always be rejected
697    /// # Important
698    /// * `energy_fn`: should be the same as used for Wang Landau, otherwise the results will be wrong!
699    pub fn entropic_step<F>(&mut self, mut energy_fn: F)
700    where
701        F: FnMut(&E) -> Option<T>,
702    {
703        unsafe { self.entropic_step_unsafe(|e| energy_fn(e)) }
704    }
705
706    /// # Accumulating entropic step
707    /// * performs a single step
708    /// # Parameter
709    /// * `energy_fn` function calculating the energy `E` of the system
710    ///   (or rather the Parameter of which you wish to obtain the probability distribution)
711    ///   during the markov steps, which can be more efficient.
712    /// # Important
713    /// * `energy_fn`: should be the same as used for Wang Landau, otherwise the results will be wrong!
714    pub fn entropic_step_acc<F>(&mut self, energy_fn: F)
715    where
716        F: FnMut(&E, &S, &mut T),
717    {
718        self.step_count += 1;
719        let step_size = self.get_stepsize();
720
721        let mut current_energy = self.old_energy.clone();
722        self.ensemble
723            .m_steps_acc(step_size, &mut self.steps, &mut current_energy, energy_fn);
724        self.entropic_step_helper(current_energy);
725    }
726
727    fn entropic_step_helper(&mut self, current_energy: T) {
728        let step_size = self.steps.len();
729
730        match self.histogram.get_bin_index(&current_energy) {
731            Ok(current_bin) => {
732                let accept_prob = self.metropolis_acception_prob(current_bin);
733
734                if self.rng.random::<f64>() > accept_prob {
735                    // reject step
736                    self.ensemble.steps_rejected(&self.steps);
737                    self.count_rejected(step_size);
738                    self.ensemble.undo_steps_quiet(&self.steps);
739                } else {
740                    // accept step
741                    self.ensemble.steps_accepted(&self.steps);
742                    self.count_accepted(step_size);
743
744                    self.old_energy = current_energy;
745                    self.old_bin = current_bin;
746                }
747            }
748            _ => {
749                // invalid step -> reject
750                self.ensemble.steps_rejected(&self.steps);
751                self.count_rejected(step_size);
752                self.ensemble.undo_steps_quiet(&self.steps);
753            }
754        };
755
756        self.histogram.increment_index(self.old_bin).unwrap();
757    }
758}

760impl<Hist, R, E, S, Res, T> Entropic for EntropicSamplingAdaptive<Hist, R, E, S, Res, T>
761where
762    Hist: Histogram,
763    R: Rng,
764{
765    /// # Number of entropic steps done until now
766    /// * will be reset by [`self.refine_estimate`](#method.refine_estimate)
767    #[inline]
768    fn step_counter(&self) -> usize {
769        self.step_count
770    }
771
772    fn total_steps_accepted(&self) -> usize {
773        self.total_entr_steps_accepted() + self.wl_steps_accepted
774    }
775
776    fn total_steps_rejected(&self) -> usize {
777        self.total_entr_steps_rejected() + self.wl_steps_rejected
778    }
779
780    /// # Number of entropic steps to be performed
781    /// * if `self` was created from `WangLandauAdaptive`,
782    ///   `step_goal` will be equal to the number of WangLandau steps, that were performed
783    #[inline]
784    fn step_goal(&self) -> usize {
785        self.step_goal
786    }
787
788    fn log_density(&self) -> Vec<f64> {
789        self.log_density_refined()
790    }
791
792    fn write_log<W: Write>(&self, mut w: W) -> Result<(), std::io::Error> {
793        writeln!(
794            w,
795            "#Acceptance prob_total: {}\n#Acceptance prob current: {}\n#total_steps: {}",
796            self.fraction_accepted_total(),
797            self.fraction_accepted_current(),
798            self.step_counter(),
799        )?;
800
801        writeln!(w, "#min_step_size {}", self.min_step_size())?;
802        writeln!(w, "#max_step_size {}", self.max_step_size())?;
803
804        write!(w, "#Current acception histogram:")?;
805        for val in self.accepted_step_hist.iter() {
806            write!(w, " {}", val)?;
807        }
808
809        write!(w, "\n#Current rejection histogram:")?;
810        for val in self.rejected_step_hist.iter() {
811            write!(w, " {}", val)?;
812        }
813
814        writeln!(w, "\n#bestof threshold: {}", self.best_of_threshold)?;
815        writeln!(w, "#min_bestof_count: {}", self.min_best_of_count)?;
816        write!(w, "\n#Current_Bestof:")?;
817
818        for val in self.best_of_steps.iter() {
819            write!(w, " {}", val)?;
820        }
821
822        write!(w, "#current_statistics_estimate:")?;
823        let estimate = self.estimate_statistics();
824        match estimate {
825            Ok(estimate) => {
826                for val in estimate {
827                    write!(w, " {}", val)?;
828                }
829                writeln!(w)
830            }
831            _ => {
832                writeln!(w, " None")
833            }
834        }
835    }
