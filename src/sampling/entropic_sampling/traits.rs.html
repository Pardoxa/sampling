3/// # Traits for quantities that all Entropic simulations have
4pub trait Entropic {
5    /// # Counter
6    /// how many Entropic steps were performed until now?
7    fn step_counter(&self) -> usize;
8
9    /// # Counter
10    /// * how many markov steps were performed until now?
11    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
12    fn steps_total(&self) -> usize {
13        self.total_steps_accepted() + self.total_steps_rejected()
14    }
15
16    /// # How many steps were accepted until now?
17    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
18    fn total_steps_accepted(&self) -> usize;
19
20    /// # How many steps were rejected until now?
21    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
22    fn total_steps_rejected(&self) -> usize;
23
24    /// # Calculate, which fraction of steps were accepted
25    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
26    /// * if no steps were performed, it returns `f64::NAN`
27    fn fraction_accepted_total(&self) -> f64 {
28        let total_acc = self.total_steps_accepted();
29        let total_steps = total_acc + self.total_steps_rejected();
30
31        if total_steps == 0 {
32            f64::NAN
33        } else {
34            total_acc as f64 / total_steps as f64
35        }
36    }
37
38    /// # Calculate, which fraction of steps were rejected
39    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
40    /// * if no steps were performed, it returns `f64::NAN`
41    fn fraction_rejected_total(&self) -> f64 {
42        let total_rej = self.total_steps_rejected();
43        let total_steps = total_rej + self.total_steps_accepted();
44
45        if total_steps == 0 {
46            f64::NAN
47        } else {
48            total_rej as f64 / total_steps as f64
49        }
50    }
51
52    /// # What is the goal to reach?
53    fn step_goal(&self) -> usize;
54
55    /// # Checks wang landau threshold
56    /// * `log_f <= log_f_threshold`
57    fn is_finished(&self) -> bool {
58        self.step_counter() >= self.step_goal()
59    }
60
61    /// # Current (non normalized) estimate of ln(P(E))
62    /// * i.e., of the natural logarithm of the
63    ///   probability density function
64    ///   for the requested interval
65    /// * this is what we are doing the simulations for
66    fn log_density(&self) -> Vec<f64>;
67
68    /// # Current (non normalized) estimate of log10(P(E))
69    /// * i.e., of logarithm with base 10 of the
70    ///   probability density function
71    ///   for the requested interval
72    /// * this is what we are doing the simulations for
73    fn log_density_base10(&self) -> Vec<f64> {
74        let mut density = self.log_density();
75        density
76            .iter_mut()
77            .for_each(|val| *val *= std::f64::consts::LOG10_E);
78        density
79    }
80
81    /// # Current (non normalized) estimate of log_base(P(E))
82    /// * i.e., of logarithm with arbitrary base of the
83    ///   probability density function
84    ///   for the requested interval
85    /// * this is what we are doing the simulations for
86    fn log_density_base(&self, base: f64) -> Vec<f64> {
87        let factor = std::f64::consts::E.log(base);
88        let mut density = self.log_density();
89        density.iter_mut().for_each(|val| *val *= factor);
90        density
91    }
92
93    /// Writes Information about the simulation to a file.
94    /// E.g. How many steps were performed.
95    fn write_log<W: Write>(&self, writer: W) -> Result<(), std::io::Error>;
96}

98/// # trait to request a reference to the current (state of the) ensemble
99/// * See also [EntropicEEH](trait.EntropicEEH.html)
100pub trait EntropicEnsemble<E>: Entropic {
101    /// return reference to current state of ensemble
102    fn ensemble(&self) -> &E;
103
104    /// # returns mutable reference to ensemble
105    /// ## Safety
106    /// * If, whatever you do with the ensemble, changes the
107    ///   energy of the current state,
108    ///   you cannot trust the results of entropic sampling anymore
109    /// * use with care
110    unsafe fn ensemble_mut(&mut self) -> &mut E;
111}

113/// # trait to request the current histogram from a Entropic simulation

123/// # trait to request the current energy from a Entropic simulation
124/// * `None` if the energy was not calculated yet
125/// * See also [EntropicEEH](trait.EntropicEEH.html)
126pub trait EntropicEnergy<Energy>: Entropic {
127    /// returns the last accepted `Energy` calculated
