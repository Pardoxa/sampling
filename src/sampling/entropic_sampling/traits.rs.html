3/// # Traits for quantities that all Entropic simulations have
4pub trait Entropic
5{
6    /// # Counter
7    /// how many Entropic steps were performed until now?
8    fn step_counter(&self) -> usize;
9
10    /// # Counter
11    /// * how many markov steps were performed until now?
12    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
13    fn steps_total(&self) -> usize{
14        self.total_steps_accepted() + self.total_steps_rejected()
15    }
16
17    /// # How many steps were accepted until now?
18    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
19    fn total_steps_accepted(&self) -> usize;
20
21    /// # How many steps were rejected until now?
22    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
23    fn total_steps_rejected(&self) -> usize;
24
25    /// # Calculate, which fraction of steps were accepted
26    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
27    /// * if no steps were performed, it returns `f64::NAN`
28    fn fraction_accepted_total(&self) -> f64 {
29        let total_acc = self.total_steps_accepted();
30        let total_steps = total_acc + self.total_steps_rejected();
31
32        if total_steps == 0 {
33            f64::NAN
34        } else {
35            total_acc as f64 / total_steps as f64
36        }
37    }
38
39    /// # Calculate, which fraction of steps were rejected
40    /// * this includes steps, that were performed to find the estimate, i.e., during wang landau
41    /// * if no steps were performed, it returns `f64::NAN`
42    fn fraction_rejected_total(&self) -> f64 {
43        let total_rej = self.total_steps_rejected();
44        let total_steps = total_rej + self.total_steps_accepted();
45
46        if total_steps == 0 {
47            f64::NAN
48        } else {
49            total_rej as f64 / total_steps as f64
50        }
51    }
52
53    /// # What is the goal to reach?
54    fn step_goal(&self) -> usize;
55    
56    /// # Checks wang landau threshold
57    /// * `log_f <= log_f_threshold`
58    fn is_finished(&self) -> bool{
59        self.step_counter() >= self.step_goal()
60    }
61    
62    /// # Current (non normalized) estimate of ln(P(E))
63    /// * i.e., of the natural logarithm of the 
64    ///   probability density function
65    ///   for the requested interval
66    /// * this is what we are doing the simulations for
67    fn log_density(&self) -> Vec<f64>;
68
69    /// # Current (non normalized) estimate of log10(P(E))
70    /// * i.e., of logarithm with base 10 of the 
71    ///   probability density function
72    ///   for the requested interval
73    /// * this is what we are doing the simulations for
74    fn log_density_base10(&self) -> Vec<f64>{
75        let mut density = self.log_density();
76        density
77            .iter_mut()
78            .for_each(|val| *val *= std::f64::consts::LOG10_E);
79        density
80        
81    }
82
83    /// # Current (non normalized) estimate of log_base(P(E))
84    /// * i.e., of logarithm with arbitrary base of the 
85    ///   probability density function
86    ///   for the requested interval
87    /// * this is what we are doing the simulations for
88    fn log_density_base(&self, base: f64) -> Vec<f64>{
89        let factor = std::f64::consts::E.log(base);
90        let mut density = self.log_density();
91        density
92            .iter_mut()
93            .for_each(|val| *val *= factor);
94        density
95    }
96
97    /// Writes Information about the simulation to a file.
98    /// E.g. How many steps were performed.
99    fn write_log<W: Write>(&self, writer: W) -> Result<(), std::io::Error>;
100    
101}

103/// # trait to request a reference to the current (state of the) ensemble 
104/// * See also [EntropicEEH](trait.EntropicEEH.html)
105pub trait EntropicEnsemble<E> : Entropic
106{
107    /// return reference to current state of ensemble
108    fn ensemble(&self) -> &E;
109
110    /// # returns mutable reference to ensemble
111    /// ## Safety
112    /// * If, whatever you do with the ensemble, changes the 
113    ///   energy of the current state, 
114    ///   you cannot trust the results of entropic sampling anymore
115    /// * use with care
116    unsafe fn ensemble_mut(&mut self) -> &mut E;
117}

119/// # trait to request the current histogram from a Entropic simulation
120/// * Note: The histogram will likely be reset multiple times during a simulation
121/// * See also [EntropicEEH](trait.EntropicEEH.html)
122pub trait EntropicHist<Hist> : Entropic
123{
124    /// # returns current histogram
125    /// * **Note**: histogram will be reset multiple times during the simulation
126    /// * please refer to the [papers](struct.EntropicAdaptive.html#adaptive-Entropic-1t)
127    fn hist(&self) -> &Hist;
128}

130/// # trait to request the current energy from a Entropic simulation
131/// * `None` if the energy was not calculated yet
132/// * See also [EntropicEEH](trait.EntropicEEH.html)
133pub trait EntropicEnergy<Energy> : Entropic
134{
135    /// returns the last accepted `Energy` calculated
136    /// `None` if no energy was calculated yet
137    fn energy(&self) -> &Energy;
138}

140/// Helper trait, so that you have to type less
141pub trait EntropicEEH<E, Hist, Energy>
