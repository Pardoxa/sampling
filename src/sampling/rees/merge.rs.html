6pub(crate) fn only_merged<Hist>(
7    merge_points: Vec<usize>,
8    alignment: Vec<usize>,
9    log_prob: Vec<Vec<f64>>,
10    e_hist: Hist,
11) -> (Vec<f64>, Hist)
12where
13    Hist: Histogram,
14{
15    let mut merged_log_prob = vec![f64::NAN; e_hist.bin_count()];
16
17    if merge_points.is_empty() {
18        merged_log_prob = match log_prob.into_iter().next() {
19            Some(prob) => prob,
20            None => {
21                //No interval present! Nothing to do!
22                return (Vec::new(), e_hist);
23            }
24        };
25        // Nothing to merge - only one interval present
26        norm_ln_prob(&mut merged_log_prob);
27        return (merged_log_prob, e_hist);
28    }
29
30    merged_log_prob[..=merge_points[0]].copy_from_slice(&log_prob[0][..=merge_points[0]]);
31
32    // https://play.rust-lang.org/?version=stable&mode=debug&edition=2018&gist=2dcb7b7a3be78397d34657ece42aa851
33    let mut align_sum = 0;
34    for (index, (&a, &mp)) in alignment.iter().zip(merge_points.iter()).enumerate() {
35        let position_l = mp + align_sum;
36        align_sum += a;
37        let left = mp - a;
38
39        let shift = merged_log_prob[position_l] - log_prob[index + 1][left];
40
41        merged_log_prob[position_l..]
42            .iter_mut()
43            .zip(log_prob[index + 1][left..].iter())
44            .for_each(|(merge, val)| {
45                *merge = val + shift;
46            });
47    }
48    norm_ln_prob(&mut merged_log_prob);
49
50    (merged_log_prob, e_hist)
51}

54pub(crate) fn align<Hist>(
55    container: &[(&[f64], &Hist)],
56) -> Result<(Vec<usize>, Vec<usize>, Vec<Vec<f64>>, Hist), HistErrors>
57where
58    Hist: HistogramCombine + Send + Sync,
59{
60    let hists: Vec<_> = container.iter().map(|v| v.1).collect();
61
62    let e_hist = Hist::encapsulating_hist(&hists)?;
63
64    let derivatives: Vec<_> = container
65        .par_iter()
66        .map(|v| derivative_merged(v.0))
67        .collect();
68
69    let alignment = hists
70        .iter()
71        .zip(hists.iter().skip(1))
72        .map(|(&left, &right)| left.align(right))
73        .collect::<Result<Vec<_>, _>>()?;
74
75    let merge_points = calc_merge_points(&alignment, &derivatives);
76
77    let log_prob = container.iter().map(|v| v.0.into()).collect();
78
79    Ok((merge_points, alignment, log_prob, e_hist))
