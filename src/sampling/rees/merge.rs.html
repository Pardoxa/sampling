9pub(crate) fn only_merged<Hist>(
10    merge_points: Vec<usize>,
11    alignment: Vec<usize>,
12    log_prob: Vec<Vec<f64>>,
13    e_hist: Hist
14) -> (Vec<f64>, Hist)
15where Hist: Histogram
16{
17    let mut merged_log_prob = vec![f64::NAN; e_hist.bin_count()];
18
19    if merge_points.is_empty() {
20        // Nothing to merge - only one interval present
21        merged_log_prob = log_prob.into_iter().next().unwrap();
22        norm_ln_prob(&mut merged_log_prob);
23        return (merged_log_prob, e_hist);
24    }
25    
26    merged_log_prob[..=merge_points[0]]
27        .copy_from_slice(&log_prob[0][..=merge_points[0]]);
28
29
30    // https://play.rust-lang.org/?version=stable&mode=debug&edition=2018&gist=2dcb7b7a3be78397d34657ece42aa851
31    let mut align_sum = 0;
32    for (index, (&a, &mp)) in alignment.iter().zip(merge_points.iter()).enumerate()
33    {
34        let position_l = mp + align_sum;
35        align_sum += a;
36        let left = mp - a;
37
38        let shift = merged_log_prob[position_l] - log_prob[index + 1][left];
39
40        merged_log_prob[position_l..]
41            .iter_mut()
42            .zip(log_prob[index + 1][left..].iter())
43            .for_each(
44                |(merge, val)|
45                {
46                    *merge = val + shift;
47                }
48            );
49
50
51    }
52    norm_ln_prob(&mut merged_log_prob);
53
54    (merged_log_prob, e_hist)
55}

60pub(crate) fn align<Hist>(container: &[(&[f64], &Hist)]) -> Result<(Vec<usize>, Vec<usize>, Vec<Vec<f64>>, Hist), HistErrors>
61where Hist: HistogramCombine + Send + Sync
62{
63    let hists: Vec<_> = container.iter()
64        .map(|v| v.1)
65        .collect();
66    
67    let e_hist = Hist::encapsulating_hist(&hists)?;
68
69    let derivatives: Vec<_> = container.par_iter()
70        .map(|v| derivative_merged(v.0))
71        .collect();
72
73    let alignment = hists.iter()
74        .zip(hists.iter().skip(1))
75        .map(|(&left, &right)| left.align(right))
76        .collect::<Result<Vec<_>, _>>()?;
77
78    let merge_points = calc_merge_points(&alignment, &derivatives);
79
80    let log_prob = container.iter()
81        .map(|v| v.0.into())
82        .collect();
83
84    Ok(
85        (
86            merge_points,
87            alignment,
88            log_prob,
89            e_hist
90        )
91    )
92}
