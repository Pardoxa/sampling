6pub(crate) fn only_merged<Hist>(
7    merge_points: Vec<usize>,
8    alignment: Vec<usize>,
9    log_prob: Vec<Vec<f64>>,
10    e_hist: Hist,
11) -> (Vec<f64>, Hist)
12where
13    Hist: Histogram,
14{
15    let mut merged_log_prob = vec![f64::NAN; e_hist.bin_count()];
16
17    if merge_points.is_empty() {
18        // Nothing to merge - only one interval present
19        merged_log_prob = log_prob.into_iter().next().unwrap();
20        norm_ln_prob(&mut merged_log_prob);
21        return (merged_log_prob, e_hist);
22    }
23
24    merged_log_prob[..=merge_points[0]].copy_from_slice(&log_prob[0][..=merge_points[0]]);
25
26    // https://play.rust-lang.org/?version=stable&mode=debug&edition=2018&gist=2dcb7b7a3be78397d34657ece42aa851
27    let mut align_sum = 0;
28    for (index, (&a, &mp)) in alignment.iter().zip(merge_points.iter()).enumerate() {
29        let position_l = mp + align_sum;
30        align_sum += a;
31        let left = mp - a;
32
33        let shift = merged_log_prob[position_l] - log_prob[index + 1][left];
34
35        merged_log_prob[position_l..]
36            .iter_mut()
37            .zip(log_prob[index + 1][left..].iter())
38            .for_each(|(merge, val)| {
39                *merge = val + shift;
40            });
41    }
42    norm_ln_prob(&mut merged_log_prob);
43
44    (merged_log_prob, e_hist)
45}

48pub(crate) fn align<Hist>(
49    container: &[(&[f64], &Hist)],
50) -> Result<(Vec<usize>, Vec<usize>, Vec<Vec<f64>>, Hist), HistErrors>
51where
52    Hist: HistogramCombine + Send + Sync,
53{
54    let hists: Vec<_> = container.iter().map(|v| v.1).collect();
55
56    let e_hist = Hist::encapsulating_hist(&hists)?;
57
58    let derivatives: Vec<_> = container
59        .par_iter()
60        .map(|v| derivative_merged(v.0))
61        .collect();
62
63    let alignment = hists
64        .iter()
65        .zip(hists.iter().skip(1))
66        .map(|(&left, &right)| left.align(right))
67        .collect::<Result<Vec<_>, _>>()?;
68
69    let merge_points = calc_merge_points(&alignment, &derivatives);
70
71    let log_prob = container.iter().map(|v| v.0.into()).collect();
72
73    Ok((merge_points, alignment, log_prob, e_hist))
74}
