36impl<Extra, Ensemble, R, Hist, Energy, S, Res> GlueAble<Hist>
37    for ReplicaExchangeEntropicSampling<Extra, Ensemble, R, Hist, Energy, S, Res>
38where
39    Hist: Clone + Histogram,
40{
41    fn push_glue_entry_ignoring(&self, job: &mut GlueJob<Hist>, ignore_idx: &[usize]) {
42        job.round_trips.extend(self.rees_roundtrip_iter());
43
44        let (hists, probs) = self.get_log_prob_and_hists();
45
46        self.walker
47            .chunks(self.chunk_size.get())
48            .zip(hists)
49            .zip(probs)
50            .enumerate()
51            .filter_map(|(index, ((walker, hist), prob))| {
52                if ignore_idx.contains(&index) {
53                    None
54                } else {
55                    Some(((walker, hist), prob))
56                }
57            })
58            .for_each(|((walker, hist), prob)| {
59                let mut missing_steps = 0;
60                let mut accepted = 0;
61                let mut rejected = 0;
62                let mut replica_exchanges = 0;
63                let mut proposed_replica_exchanges = 0;
64                for w in walker {
65                    if !w.is_finished() {
66                        let missing = w.step_threshold().saturating_sub(w.step_count());
67                        if missing > missing_steps {
68                            missing_steps = missing;
69                        }
70                    }
71
72                    let r = w.rejected_markov_steps();
73                    let a = w.step_count() - r;
74                    rejected += r;
75                    accepted += a;
76                    replica_exchanges += w.replica_exchanges();
77                    proposed_replica_exchanges += w.proposed_replica_exchanges();
78                }
79
80                let stats = IntervalSimStats {
81                    sim_progress: SimProgress::MissingSteps(missing_steps),
82                    interval_sim_type: SimulationType::REES,
83                    rejected_steps: rejected,
84                    accepted_steps: accepted,
85                    replica_exchanges: Some(replica_exchanges),
86                    proposed_replica_exchanges: Some(proposed_replica_exchanges),
87                    merged_over_walkers: self.chunk_size,
88                };
89
90                job.collection.push(GlueEntry {
91                    hist: hist.clone(),
92                    prob,
93                    log_base: LogBase::BaseE,
94                    interval_stats: stats,
95                });
96            })
97    }
98}

399    pub(crate) fn update_roundtrips(&mut self) {
400        if self.num_intervals() <= 1 {
401            return;
402        }
403        // check all walker that are currently in the first interval
404        let mut chunk_iter = self.walker.chunks(self.chunk_size.get());
405        let first_chunk = chunk_iter.next().unwrap();
406        first_chunk.iter().for_each(|walker| {
407            let id = walker.id();
408            let last_visited = match self.rees_last_extreme_interval_visited.get_mut(id) {
409                Some(last) => last,
410                None => unreachable!(),
411            };
412
413            match last_visited {
414                ExtremeInterval::Right => {
415                    *last_visited = ExtremeInterval::Left;
416                    self.rees_roundtrip_halfes[id] += 1;
417                }
418                ExtremeInterval::None => {
419                    *last_visited = ExtremeInterval::Left;
420                }
421                _ => (),
422            }
423        });
424
425        // check all walker that are currently in the last interval
426        let last_chunk = match chunk_iter.last() {
427            Some(chunk) => chunk,
428            None => unreachable!(),
429        };
430
431        last_chunk.iter().for_each(|walker| {
432            let id = walker.id();
433            let last_visited = match self.rees_last_extreme_interval_visited.get_mut(id) {
434                Some(last) => last,
435                None => unreachable!(),
436            };
437
438            match last_visited {
439                ExtremeInterval::Left => {
440                    *last_visited = ExtremeInterval::Right;
441                    self.rees_roundtrip_halfes[id] += 1;
442                }
443                ExtremeInterval::None => {
444                    *last_visited = ExtremeInterval::Right;
445                }
446                _ => (),
447            }
448        });
449    }

478impl<Extra, Ensemble, R, Hist, Energy, S, Res> Rees<Extra, Ensemble, R, Hist, Energy, S, Res>
479where
480    Ensemble: Send + Sync + MarkovChain<S, Res>,
481    R: Send + Sync + Rng,
482    Extra: Send + Sync,
483    Hist: Send + Sync + Histogram + HistogramVal<Energy>,
484    Energy: Send + Sync + Clone,
485    S: Send + Sync,
486    Res: Send + Sync,
487{
488    /// # Refine the estimate of the probability density functions
489    /// * refines the estimate of all walkers
490    /// * does so by calling the walker method [refine](`crate::rees::ReesWalker::refine`)
491    pub fn refine(&mut self) {
492        self.walker.par_iter_mut().for_each(|w| w.refine());
493    }
494
495    /// # Sweep
496    /// * Performs one sweep of the Replica exchange entropic sampling simulation
497    /// * You can make a complete simulation, by repeatedly calling this method
498    ///   until `self.is_finished()` returns true
499    pub fn sweep<F, P>(&mut self, energy_fn: F, extra_fn: P)
500    where
501        F: Fn(&mut Ensemble) -> Option<Energy> + Copy + Send + Sync,
502        P: Fn(&ReesWalker<R, Hist, Energy, S, Res>, &mut Ensemble, &mut Extra) + Copy + Send + Sync,
503    {
504        let slice = self.ensembles.as_slice();
505
506        #[cfg(not(feature = "sweep_time_optimization"))]
507        let walker = &mut self.walker;
508
509        #[cfg(feature = "sweep_time_optimization")]
510        let mut walker = {
511            let mut walker = Vec::with_capacity(self.walker.len());
512            walker.extend(self.walker.iter_mut());
513            walker.par_sort_unstable_by_key(|w| Reverse(w.duration()));
514            walker
515        };
516
517        walker
518            .par_iter_mut()
519            .zip(self.extra.par_iter_mut())
520            .for_each(|(w, extra)| w.sweep(slice, extra, extra_fn, energy_fn));
521
522        // replica exchange
523        if self.walkers_per_interval().get() > 1 {
524            let exchange_m = self.replica_exchange_mode;
525
526            self.walker
527                .par_chunks_mut(self.chunk_size.get())
528                .for_each(|chunk| {
529                    let mut shuf = Vec::with_capacity(chunk.len());
530                    if let Some((first, rest)) = chunk.split_first_mut() {
531                        shuf.extend(rest.iter_mut());
532                        shuf.shuffle(&mut first.rng);
533                        shuf.push(first);
534                        let s = if exchange_m {
535                            &mut shuf
536                        } else {
537                            &mut shuf[1..]
538                        };
539
540                        s.chunks_exact_mut(2).for_each(|c| {
541                            let ptr = c.as_mut_ptr();
542                            unsafe {
543                                let a = &mut *ptr;
544                                let b = &mut *ptr.offset(1);
545                                replica_exchange(a, b);
546                            }
547                        });
548                    }
549                });
550        }
551
552        let walker_slice = if self.replica_exchange_mode {
553            &mut self.walker
554        } else {
555            &mut self.walker[self.chunk_size.get()..]
556        };
557        self.replica_exchange_mode = !self.replica_exchange_mode;
558
559        let chunk_size = self.chunk_size;
560
561        walker_slice
562            .par_chunks_exact_mut(2 * self.chunk_size.get())
563            .for_each(|walker_chunk| {
564                let (slice_a, slice_b) = walker_chunk.split_at_mut(chunk_size.get());
565
566                let mut slice_b_shuffled: Vec<_> = slice_b.iter_mut().collect();
567                slice_b_shuffled.shuffle(&mut slice_a[0].rng);
568
569                for (walker_a, walker_b) in slice_a.iter_mut().zip(slice_b_shuffled.into_iter()) {
570                    replica_exchange(walker_a, walker_b);
571                }
572            });
573
574        self.update_roundtrips()
575    }

577    /// # Perform the Replica exchange simulation
578    /// * will simulate until **all** walkers are finished
579    /// * extra_fn should be used for example for writing Data to a file
580    pub fn simulate_until_convergence<F, P>(&mut self, energy_fn: F, extra_fn: P)
581    where
582        Ensemble: Send + Sync,
583        R: Send + Sync,
584        F: Fn(&mut Ensemble) -> Option<Energy> + Copy + Send + Sync,
585        P: Fn(&ReesWalker<R, Hist, Energy, S, Res>, &mut Ensemble, &mut Extra) + Copy + Send + Sync,
586    {
587        while !self.is_finished() {
588            self.sweep(energy_fn, extra_fn);
589        }
590    }
591
592    /// # Perform the Replica exchange simulation
593    /// * will simulate until **all** walkers are finished **or**
594    /// * until condition returns false
595    pub fn simulate_while<F, C, P>(&mut self, energy_fn: F, mut condition: C, extra_fn: P)
596    where
597        Ensemble: Send + Sync,
598        R: Send + Sync,
599        F: Fn(&mut Ensemble) -> Option<Energy> + Copy + Send + Sync,
600        C: FnMut(&Self) -> bool,
601        P: Fn(&ReesWalker<R, Hist, Energy, S, Res>, &mut Ensemble, &mut Extra) + Copy + Send + Sync,
602    {
603        while !self.is_finished() && condition(self) {
604            self.sweep(energy_fn, extra_fn);
605        }
606    }

608    /// # Sanity check
609    /// * checks if the stored (i.e., last) energy(s) of the system
610    ///   match with the result of energy_fn
611    pub fn check_energy_fn<F>(&mut self, energy_fn: F) -> bool
612    where
613        Energy: PartialEq,
614        F: Fn(&mut Ensemble) -> Option<Energy> + Copy + Send + Sync,
615    {
616        let ensembles = self.ensembles.as_slice();
617        self.walker
618            .par_iter()
619            .all(|w| w.check_energy_fn(ensembles, energy_fn))
620    }

622    #[allow(clippy::type_complexity)]
623    #[deprecated]
624    fn merged_log_probability_helper(
625        &self,
626    ) -> Result<(Vec<usize>, Vec<usize>, Vec<Vec<f64>>, Hist), HistErrors>
627    where
628        Hist: HistogramCombine,
629    {
630        // get the log_probabilities - the walkers over the same intervals are merged
631        let mut log_prob: Vec<_> = self
632            .walker
633            .par_chunks(self.chunk_size.get())
634            .map(get_merged_refined_walker_prob)
635            .collect();
636
637        log_prob.par_iter_mut().for_each(|v| {
638            subtract_max(v);
639        });
640
641        // get the derivative, for merging later
642        let derivatives: Vec<_> = log_prob.par_iter().map(|v| derivative_merged(v)).collect();
643
644        let hists: Vec<_> = self
645            .walker
646            .iter()
647            .step_by(self.chunk_size.get())
648            .map(|w| w.hist())
649            .collect();
650
651        let e_hist = Hist::encapsulating_hist(&hists)?;
652
653        let alignment = hists
654            .iter()
655            .zip(hists.iter().skip(1))
656            .map(|(&left, &right)| left.align(right))
657            .collect::<Result<Vec<_>, _>>()?;
658
659        let merge_points: Vec<_> = calc_merge_points(&alignment, &derivatives);
660
661        Ok((merge_points, alignment, log_prob, e_hist))
662    }

688    fn get_glue_stats(&self) -> GlueStats {
689        let stats = self
690            .walker
691            .chunks(self.chunk_size.get())
692            .map(|walker| {
693                let mut missing_steps = 0;
694                let mut accepted = 0;
695                let mut rejected = 0;
696                let mut replica_exchanges = 0;
697                let mut proposed_replica_exchanges = 0;
698                for w in walker {
699                    if !w.is_finished() {
700                        let missing = w.step_threshold() - w.step_count();
701                        if missing > missing_steps {
702                            missing_steps = missing;
703                        }
704                    }
705
706                    let r = w.rejected_markov_steps();
707                    let a = w.step_count() - r;
708                    rejected += r;
709                    accepted += a;
710                    replica_exchanges += w.replica_exchanges();
711                    proposed_replica_exchanges += w.proposed_replica_exchanges();
712                }
713
714                IntervalSimStats {
715                    sim_progress: SimProgress::MissingSteps(missing_steps),
716                    interval_sim_type: SimulationType::REES,
717                    rejected_steps: rejected,
718                    accepted_steps: accepted,
719                    replica_exchanges: Some(replica_exchanges),
720                    proposed_replica_exchanges: Some(proposed_replica_exchanges),
721                    merged_over_walkers: self.chunk_size,
722                }
723            })
724            .collect();
725        let roundtrips = self.rees_roundtrip_iter().collect();
726        GlueStats {
727            interval_stats: stats,
728            roundtrips,
729        }
730    }

732    /// # Results of the simulation
733    ///
734    /// This is what we do the simulation for!
735    ///
736    /// It returns [Glued] which allows you to print out the merged probability density function.
737    /// It also allows you to switch the base of the logarithm and so on, have a look!
738    ///
739    /// It will use an average based merging algorithm, i.e., it will try to align the intervals
740    /// and merge them by using the values obtained by averaging in log-space
741    ///
742    /// ## Notes
743    /// Fails if the internal histograms (intervals) do not align. Might fail if
744    /// there is no overlap between neighboring intervals
745    pub fn average_merged_log_probability_and_align(
746        &self,
747    ) -> Result<Glued<Hist, Energy>, HistErrors>
748    where
749        Hist: HistogramCombine,
750    {
751        let (hists, log_probs) = self.get_log_prob_and_hists();
752
753        let mut res = average_merged_and_aligned(log_probs, &hists, LogBase::BaseE)?;
754        let stats = self.get_glue_stats();
755        res.set_stats(stats);
756        Ok(res)
757    }

759    /// # Results of the simulation
760    ///
761    /// This is what we do the simulation for!
762    ///
763    /// It returns [Glued] which allows you to print out the merged probability density function.
764    /// It also allows you to switch the base of the logarithm and so on, have a look!
765    ///
766    /// It will use an derivative based merging algorithm, i.e., it will try to align the intervals
767    /// and merge them by looking at the derivatives of the probability density function.
768    /// It will search for the (merging-)point where the derivatives are the most similar to each other
769    /// and glue by using the values of one of the intervals before the merging point and the
770    /// other interval afterwards. This is repeated for every interval
771    ///
772    /// ## Notes
773    /// Fails if the internal histograms (intervals) do not align. Might fail if
774    /// there is no overlap between neighboring intervals
775    pub fn derivative_merged_log_prob_and_aligned(&self) -> Result<Glued<Hist, Energy>, HistErrors>
776    where
777        Hist: HistogramCombine + Histogram,
778    {
779        let (hists, log_probs) = self.get_log_prob_and_hists();
780
781        let mut res = derivative_merged_and_aligned(log_probs, &hists, LogBase::BaseE)?;
782        let stats = self.get_glue_stats();
783        res.set_stats(stats);
784        Ok(res)
785    }

788        since = "0.2.0",
789        note = "will be removed in future releases. Use new method 'derivative_merged_log_prob_and_aligned' or consider using 'average_merged_log_probability_and_align' instead"

793    where
794        Hist: HistogramCombine,
795    {
796        let (merge_points, alignment, log_prob, e_hist) = self.merged_log_probability_helper()?;
797        Ok(only_merged(merge_points, alignment, log_prob, e_hist))
798    }

817/// # Merge probability density of multiple rees simulations
818/// * very similar to [merged_log_prob](`crate::rees::Rees::merged_log_prob_rees`)
819///
820/// The difference is, that this function will ignore the specified walkers,
821/// therefore `ignore` should be a slice of indices, which are to be ignored.
822/// The slice does not have to be sorted in any way, though duplicate indices
823/// and indices which are out of bounds will be ignored for the ignore list
824pub fn merged_log_prob_ignore<Extra, Ensemble, R, Hist, Energy, S, Res>(
825    rees: &[Rees<Extra, Ensemble, R, Hist, Energy, S, Res>],
826    ignore: &[usize],
827) -> Result<(Vec<f64>, Hist), HistErrors>
828where
829    Hist: HistogramVal<Energy> + HistogramCombine + Histogram + Send + Sync,
830    Energy: PartialOrd,
831{
832    if rees.is_empty() {
833        return Err(HistErrors::EmptySlice);
834    }
835    let merged_prob = merged_probs(rees);
836    let mut container = combine_container(rees, &merged_prob);
837    ignore_fn(&mut container, ignore);
838    let (merge_points, alignment, log_prob, e_hist) = align(&container)?;
839    Ok(only_merged(merge_points, alignment, log_prob, e_hist))
840}

861fn combine_container<'a, Ensemble, R, Hist, Energy, S, Res, Extra>(
862    rees: &'a [Rees<Extra, Ensemble, R, Hist, Energy, S, Res>],
863    merged_probs: &'a [Vec<f64>],
864) -> Vec<(&'a [f64], &'a Hist)>
865where
866    Hist: HistogramVal<Energy> + HistogramCombine,
867    Energy: PartialOrd,
868{
869    let hists: Vec<_> = rees
870        .iter()
871        .flat_map(|r| {
872            r.walkers()
873                .iter()
874                .step_by(r.walkers_per_interval().get())
875                .map(|w| w.hist())
876        })
877        .collect();
878
879    assert_eq!(hists.len(), merged_probs.len());
880
881    let mut container: Vec<_> = merged_probs
882        .iter()
883        .zip(hists)
884        .map(|(prob, hist)| (prob.as_slice(), hist))
885        .collect();
886
887    container.sort_unstable_by(|a, b| {
888        a.1.first_border()
889            .partial_cmp(&b.1.first_border())
890            .unwrap_or(Ordering::Equal)
891    });
892    container
893}

895fn merged_probs<Ensemble, R, Hist, Energy, S, Res, Extra>(
896    rees: &[Rees<Extra, Ensemble, R, Hist, Energy, S, Res>],
897) -> Vec<Vec<f64>>
898where
899    Hist: Histogram,
900{
901    let merged_probs: Vec<_> = rees
902        .iter()
903        .flat_map(|rees| {
904            rees.walkers()
905                .chunks(rees.walkers_per_interval().get())
906                .map(get_merged_walker_prob)
907        })
908        .collect();
909    merged_probs
910}
911
912fn get_merged_walker_prob<R, Hist, Energy, S, Res>(
913    walker: &[ReesWalker<R, Hist, Energy, S, Res>],
914) -> Vec<f64>
915where
916    Hist: Histogram,
917{
918    let log_len = walker[0].log_density().len();
919    debug_assert!(walker.iter().all(|w| w.log_density().len() == log_len));
920
921    let mut averaged_log_density = walker[0].log_density_refined();
922    norm_ln_prob(&mut averaged_log_density);
923
924    if walker.len() > 1 {
925        walker[1..].iter().for_each(|w| {
926            let mut density = w.log_density_refined();
927            norm_ln_prob(&mut density);
928            averaged_log_density
929                .iter_mut()
930                .zip(density)
931                .for_each(|(average, other)| {
932                    *average += other;
933                })
934        });
935
936        let number_of_walkers = walker.len() as f64;
937        averaged_log_density
938            .iter_mut()
939            .for_each(|average| *average /= number_of_walkers);
940    }
941
942    averaged_log_density
943}
