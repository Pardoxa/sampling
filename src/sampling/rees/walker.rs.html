186    /// * Current non normalized estimate of the natural logarithm of the probability density function
187    /// * calculated by refining old density with current histogram
188    /// 
189    /// # How does the refining work?
190    /// * Let P(i) be the current probability density function (non normalized) at some index i
191    /// * Let H(i) be the histogram at some index i
192    /// 
193    /// We will now calculate the refined density P', which is calculated as follows:
194    /// 
195    /// P'(i) = P(i) * H(i) (if H(i) != 0)
196    ///
197    /// P'(i) = P(i) (if H(i) == 0)
198    ///
199    /// Or in log space, which is what is actually calculated here:
200    ///
201    /// ln(P'(i)) = ln(P(i)) + ln(H(i)) (if H(i) != 0)
202    ///
203    /// ln(P'(i)) = ln(P(i)) (if H(i)=0)
204    ///
205    /// # for more information see
206    /// > J. Lee,
207    /// > “New Monte Carlo algorithm: Entropic sampling,”
208    /// > Phys. Rev. Lett. 71, 211–214 (1993),
209    /// > DOI: [10.1103/PhysRevLett.71.211](https://doi.org/10.1103/PhysRevLett.71.211)
210    pub fn log_density_refined(&self) -> Vec<f64>
211    where Hist: Histogram
212    {
213        let mut refined_log_density = Vec::with_capacity(self.log_density.len());
214
215        refined_log_density.extend(
216            self.log_density
217                .iter()
218                .zip(self.hist.hist().iter())
219                .map(
220                    |(&log_d, &h)|
221                    {
222                        if h == 0 {
223                            log_d
224                        } else {
225                            log_d + (h as f64).ln()
226                        }
227                    }
228                )
229        ); 
230        refined_log_density
231    }
232
233    /// # Old estimate of log10 of probability density

289    pub(crate) fn check_energy_fn<F, Ensemble>(
290        &self,
291        ensemble_vec: &[RwLock<Ensemble>],
292        energy_fn: F
293    )   -> bool
294    where Energy: PartialEq,F: Fn(&mut Ensemble) -> Option<Energy>,
295        
296    {
297        let mut e = ensemble_vec[self.id]
298            .write()
299            .expect("Fatal Error encountered; ERRORCODE 0x1 - this should be \
300                impossible to reach. If you are using the latest version of the \
301                'sampling' library, please contact the library author via github by opening an \
302                issue! https://github.com/Pardoxa/sampling/issues");
303        
304        let energy = match energy_fn(&mut e){
305            Some(energy) => energy,
306            None => {
307                return false;
308            }
309        };
310        energy == self.old_energy
311    }
312
313    pub(crate) fn sweep<Ensemble, F, Extra, P>
314    (
315        &mut self,
316        ensemble_vec: &[RwLock<Ensemble>],
317        extra: &mut Extra,
318        extra_fn: P,
319        energy_fn: F,
320    )
321    where F: Fn(&mut Ensemble) -> Option<Energy>,
322        P: Fn(&Self, &mut Ensemble, &mut Extra),
323        Ensemble: MarkovChain<S, Res>,
324        Hist: Histogram
325    {
326        #[cfg(feature = "sweep_time_optimization")]
327        let start = Instant::now();
328
329        let mut e = ensemble_vec[self.id]
330            .write()
331            .expect("Fatal Error encountered; ERRORCODE 0x3 - this should be \
332                impossible to reach. If you are using the latest version of the \
333                'sampling' library, please contact the library author via github by opening an \
334                issue! https://github.com/Pardoxa/sampling/issues");
335        
336        for _ in 0..self.sweep_size.get()
337        {   
338            self.step_count += 1;
339            e.m_steps(self.step_size, &mut self.markov_steps);
340
341            let energy = match energy_fn(&mut e){
342                Some(energy) => energy,
343                None => {
344                    self.count_rejected();
345                    e.undo_steps_quiet(&self.markov_steps);
346                    continue;
347                }
348            };
349
350
351            match self.hist.get_bin_index(&energy) 
352            {
353                Ok(current_bin) => {
354                    // metropolis hastings
355                    let acception_prob = (self.log_density[self.bin] - self.log_density[current_bin])
356                        .exp();
357                    if self.rng.random::<f64>() > acception_prob 
358                    {
359                        self.count_rejected();
360                        e.steps_rejected(&self.markov_steps);
361                        e.undo_steps_quiet(&self.markov_steps);
362                    } else {
363                        self.old_energy = energy;
364                        self.bin = current_bin;
365                        e.steps_accepted(&self.markov_steps);
366                    }
367                },
368                _ => {
369                    self.count_rejected();
370                    e.steps_rejected(&self.markov_steps);
371                    e.undo_steps_quiet(&self.markov_steps);
372                }
373            }
374
375            self.hist.increment_index(self.bin)
376                .expect("Histogram index Error, ERRORCODE 0x4");
377
378            extra_fn(self, &mut e, extra);
379
380        }
381
382        #[cfg(feature = "sweep_time_optimization")]
383            {
384                self.duration = start.elapsed();
385            }
386    }
387}

389pub(crate) fn replica_exchange<R, Hist, Energy, S, Res>
390(
391    walker_a: &mut ReesWalker<R, Hist, Energy, S, Res>,
392    walker_b: &mut ReesWalker<R, Hist, Energy, S, Res>
393) where Hist: HistogramVal<Energy> + Histogram,
394    R: Rng
395{
396    walker_a.proposed_re += 1;
397    walker_b.proposed_re += 1;
398    // check if exchange is even possible
399    let new_bin_a = match walker_a.hist.get_bin_index(&walker_b.old_energy)
400    {
401        Ok(bin) => bin,
402        _ => return,
403    };
404
405    let new_bin_b = match walker_b.hist.get_bin_index(&walker_a.old_energy)
406    {
407        Ok(bin) => bin,
408        _ => return,
409    };
410
411    // see paper equation 1
412    let log_gi_x = walker_a.log_density[walker_a.bin];
413    let log_gi_y = walker_a.log_density[new_bin_a];
414
415    let log_gj_y = walker_b.log_density[walker_b.bin];
416    let log_gj_x = walker_b.log_density[new_bin_b];
417
418    let log_prob = log_gi_x + log_gj_y - log_gi_y - log_gj_x;
419
420    let prob = log_prob.exp();
421
422    // if exchange is accepted
423    if walker_b.rng.random::<f64>() < prob 
424    {
425        swap(&mut walker_b.id, &mut walker_a.id);
426        swap(&mut walker_b.old_energy, &mut walker_a.old_energy);
427        walker_b.bin = new_bin_b;
428        walker_a.bin = new_bin_a;
429        walker_a.re += 1;
430        walker_b.re += 1;
431    }
432
433    walker_a.hist.increment_index(walker_a.bin).unwrap();
434    walker_b.hist.increment_index(walker_b.bin).unwrap();
435}

437pub(crate) fn get_merged_refined_walker_prob<R, Hist, Energy, S, Res>(walker: &[ReesWalker<R, Hist, Energy, S, Res>]) -> Vec<f64>
438where Hist: Histogram
439{
440    let log_len = walker[0].log_density.len();
441    debug_assert!(
442        walker.iter()
443            .all(|w| w.log_density.len() == log_len)
444    );
445
446    let mut averaged_log_density = walker[0].log_density_refined();
447
448    if walker.len() > 1 {
449
450        walker[1..]
451            .iter()
452            .map(|w| w.log_density_refined())
453            .for_each(
454                |log_density|
455                {
456                    averaged_log_density.iter_mut()
457                        .zip(log_density)
458                        .for_each(
459                            |(average, other)|
460                            {
461                                *average += other;
462                            }
463                        )
464                }
465            );
466    
467        let number_of_walkers = walker.len() as f64;
468        averaged_log_density.iter_mut()
469            .for_each(|average| *average /= number_of_walkers);
470    }
471
472    averaged_log_density
