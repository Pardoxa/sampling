2/// # Create a markov chain by doing markov steps
3pub trait MarkovChain<S, Res> {
4    /// * undo a markov step, return result-state
5    /// * if you want to undo more than one step
6    ///   see [`undo_steps`](#method.undo_steps)
7    fn undo_step(&mut self, step: &S) -> Res;
8
9    /// * undo a markov, **panic** on invalid result state
10    /// * for undoing multiple steps see [`undo_steps_quiet`](#method.undo_steps_quiet)
11    fn undo_step_quiet(&mut self, step: &S);
12
13    /// # Markov step
14    /// * use this to perform a markov step step
15    /// * for doing multiple markov steps at once, use [`m_steps`](#method.m_steps)
16    fn m_step(&mut self) -> S;
17
18    /// #  Markov steps
19    /// * use this to perform multiple markov steps at once
20    /// * steps can be used to undo the steps with `self.undo_steps(steps)`
21    /// * `steps` will be emptied before step is performed
22    #[inline]
23    fn m_steps(&mut self, count: usize, steps: &mut Vec<S>) {
24        steps.clear();
25        steps.extend((0..count)
26            .map(|_| self.m_step())
27        );
28    }
29
30    /// # Markov steps without return
31    /// * use this to perform multiple markov steps at once
32    /// * only use this if you **know** that you do **not** want to undo the steps
33    /// * you cannot undo this steps, but therefore it does not need to allocate a vector 
34    ///   for undoing steps
35    fn m_steps_quiet(&mut self, count: usize)
36    {
37        for _ in 0..count {
38            self.m_step();
39        }
40    }
41
42    /// # Accumulating markov step
43    /// * this calculates something while performing the markov chain, e.g., the current energy,
44    ///   which can be more efficient then calculating it from scratch afterwards
45    #[inline]
46    fn m_step_acc<Acc, AccFn>(&mut self, acc: &mut Acc, mut acc_fn: AccFn) -> S
47    where AccFn: FnMut(&Self, &S, &mut Acc)
48    {
49        let s = self.m_step();
50        acc_fn(self, &s, acc);
51        s
52    }
53
54    /// # Accumulating markov steps
55    /// * this calculates something while performing the markov chain, e.g., the current energy
56    ///   which can be more efficient then calculating it from scratch afterwards
57    #[inline]
58    fn m_steps_acc<Acc, AccFn>
59    (
60        &mut self,
61        count: usize,
62        steps: &mut Vec<S>,
63        acc: &mut Acc,
64        mut acc_fn: AccFn
65    )
66    where AccFn: FnMut(&Self, &S, &mut Acc)
67    {
68        steps.clear();
69        steps.extend(
70            (0..count)
71                .map(|_| self.m_step_acc(acc, &mut acc_fn))
72        );
73    }
74
75    /// # Accumulating markov steps
76    /// * this calculates something while performing the markov chain, e.g., the current energy
77    ///   which can be more efficient then calculating it from scratch afterwards
78    /// * quiet step, i.e., you will not be able to undo the step
79    #[inline]
80    fn m_steps_acc_quiet<Acc, AccFn>(&mut self, count: usize, acc: &mut Acc, mut acc_fn: AccFn)
81    where AccFn: FnMut(&Self, &S, &mut Acc)
82    {
83        for _ in 0..count{
84            let _ = self.m_step_acc(acc, &mut acc_fn);
85        }
86    }
87
88    /// # Undo markov steps
89    /// * Note: uses undo_step in correct order and returns result
90    /// ## Important:
91    /// * look at specific implementation of `undo_step`, every thing mentioned there applies to each step
92    fn undo_steps(&mut self, steps: &[S], res: &mut Vec<Res>) {
93        res.clear();
94        res.extend(
95            steps.iter()
96                .rev()
97                .map(|step| self.undo_step(step))
98        );
99    }
100
101    /// # Undo markov steps
102    /// * Note: uses `undo_step_quiet` in correct order
103    /// ## Important:
104    /// * look at specific implementation of `undo_step_quiet`, every thing mentioned there applies to each step
105    fn undo_steps_quiet(&mut self, steps: &[S]) {
106        steps.iter()
107            .rev()
108            .for_each( |step| self.undo_step_quiet(step));
109    }
110
111
112    /// # Function called whenever the steps are accepted.
113    /// *You can use it to create the acceptance statistics if 
114    /// you move a variety of different moves
115    /// * If you use the default implementation this will be a optimized out
116    #[inline]
117    fn steps_accepted(&mut self, _steps: &[S])
118    {
119
120    }
121
122    /// # Function called whenever the steps are rejected.
123    /// * You can use it to create the acceptance statistics if 
124    ///   you move a variety of different moves
125    /// * If you use the default implementation this will be a optimized out
126    #[inline]
127    fn steps_rejected(&mut self, _steps: &[S])
128    {
129
130    }
131}

133/// For easy sampling of your ensemble
134pub trait SimpleSample{
135    /// # Randomizes self according to  model
136    /// * this is intended for creation of initial sample
137    /// * used in [`simple_sample`](#method.simple_sample)
138    ///   and [`simple_sample_vec`](#method.simple_sample_vec)
139    fn randomize(&mut self);
140
141    /// # do the following `times` times:
142    /// 1) `f(self)`
143    /// 2) `self.randomize()`
144    fn simple_sample<F>(&mut self, times: usize, mut f: F)
145        where F: FnMut(&Self)
146    {
147        for _ in 0..times {
148            f(self);
149            self.randomize();
150        }
151    }
152
153    /// # do the following `times` times:
154    /// 1) `res = f(self)`
155    /// 2) `self.randomize()`
156    /// ## res is collected into Vector
157    fn simple_sample_vec<F, G>(&mut self, times: usize, mut f: F) -> Vec<G>
158        where F: FnMut(&Self) -> G
159    {
160        let mut vec = Vec::with_capacity(times);
161        for _ in 0..times {
162            vec.push(f(self));
163            self.randomize();
164        }
165        vec
166    }
167}
