9/// # Result of flipping a coin
10pub enum CoinFlip {
11    /// The result is Head
12    Head,
13    /// The result is Tail
14    Tail,
15}
16
17impl CoinFlip {
18    /// Turn Coin around, i.e., invert CoinFlip
19    pub fn turn(&mut self) {
20        *self = match self {
21            CoinFlip::Head => CoinFlip::Tail,
22            CoinFlip::Tail => CoinFlip::Head,
23        };
24    }
25}

37/// # A sequence of Coin flips. Contains random Number generator
38pub struct CoinFlipSequence<R> {
39    rng: R,
40    seq: Vec<CoinFlip>,
41    /// You can ignore everything after here, it is just used for testing
42    steps: usize,
43    rejected: usize,
44    accepted: usize,
45    undo_count: usize,
46}
47
48impl<R> CoinFlipSequence<R>
49where
50    R: Rng,
51{
52    /// Create new coin flip sequence
53    /// * length `n`
54    /// * use `rng` as random number generator
55    pub fn new(n: usize, mut rng: R) -> Self {
56        let mut seq = Vec::with_capacity(n);
57        seq.extend((0..n).map(|_| {
58            if rng.random::<bool>() {
59                CoinFlip::Tail
60            } else {
61                CoinFlip::Head
62            }
63        }));
64        Self {
65            rng,
66            seq,
67            steps: 0,
68            rejected: 0,
69            accepted: 0,
70            undo_count: 0,
71        }
72    }
73}
74
75impl<R> CoinFlipSequence<R> {
76    /// Count how often `Head` occurs in the Coin flip sequence
77    pub fn head_count(&self) -> usize {
78        self.seq
79            .iter()
80            .filter(|&item| *item == CoinFlip::Head)
81            .count()
82    }
83
84    /// * Calculate the head count, if a previous head count of the ensemble and the
85    ///   markov steps leading to the current state are known
86    /// * `head_count` is updated
87    /// * might **panic** if `step` was not the markov step leading from the ensemble with `head_count`
88    ///   to the current ensemble - if it does not panic, the result will be wrong
89    pub fn update_head_count(&self, step: &CoinFlipMove, head_count: &mut usize) {
90        match step.previous {
91            CoinFlip::Head => {
92                *head_count -= 1;
93            }
94            CoinFlip::Tail => {
95                *head_count += 1;
96            }
97        }
98    }
99
100    /// Count many times `Head` occurred in a row
101    /// * uses maximum value, i.e., for the sequence `HHTHHHT` it will return 3
102    pub fn max_heads_in_a_row(&self) -> usize {
103        let mut current_heads = 0;
104        let mut max_heads = 0;
105        for flip in self.seq.iter() {
106            match flip {
107                CoinFlip::Head => current_heads += 1,
108                CoinFlip::Tail => {
109                    max_heads = max_heads.max(current_heads);
110                    current_heads = 0;
111                }
112            }
113        }
114        max_heads.max(current_heads)
115    }
116}

118impl<R> MarkovChain<CoinFlipMove, ()> for CoinFlipSequence<R>
119where
120    R: Rng,
121{
122    /// Perform a markov step
123    fn m_step(&mut self) -> CoinFlipMove {
124        // draw a random position
125        let pos = self.rng.random_range(0..self.seq.len());
126        let previouse = self.seq[pos];
127        // flip coin at that position
128        self.seq[pos].turn();
129        // information to restore the previouse state
130        CoinFlipMove {
131            previous: previouse,
132            index: pos,
133        }
134    }
135
136    /// # Only implemented for testcases
137    /// Default implementation would suffice
138    #[inline]
139    fn m_steps(&mut self, count: usize, steps: &mut Vec<CoinFlipMove>) {
140        self.steps += 1;
141        steps.clear();
142        steps.extend((0..count).map(|_| self.m_step()));
143    }
144
145    /// # Only implemented for testcases
146    /// Default implementation would suffice
147    #[inline]
148    fn m_steps_acc<Acc, AccFn>(
149        &mut self,
150        count: usize,
151        steps: &mut Vec<CoinFlipMove>,
152        acc: &mut Acc,
153        mut acc_fn: AccFn,
154    ) where
155        AccFn: FnMut(&Self, &CoinFlipMove, &mut Acc),
156    {
157        self.steps += 1;
158        steps.clear();
159        steps.extend((0..count).map(|_| self.m_step_acc(acc, &mut acc_fn)));
160    }
161
162    fn undo_step(&mut self, step: &CoinFlipMove) {
163        self.seq[step.index] = step.previous;
164    }
165
166    #[inline]
167    fn undo_step_quiet(&mut self, step: &CoinFlipMove) {
168        self.undo_step(step);
169    }
170
171    /// # Only implemented for testcases
172    /// Default implementation would suffice
173    fn undo_steps(&mut self, steps: &[CoinFlipMove], res: &mut Vec<()>) {
174        self.undo_count += 1;
175        res.clear();
176        res.extend(steps.iter().rev().map(|step| self.undo_step(step)));
177        assert_eq!(self.rejected, self.undo_count);
178    }
179
180    /// # Only implemented for testcases
181    /// Default implementation would suffice
182    fn undo_steps_quiet(&mut self, steps: &[CoinFlipMove]) {
183        self.undo_count += 1;
184        steps
185            .iter()
186            .rev()
187            .for_each(|step| self.undo_step_quiet(step));
188        assert_eq!(self.rejected, self.undo_count);
189    }
190
191    /// # Only implemented for testcases
192    /// Default implementation would suffice
193    fn steps_accepted(&mut self, _steps: &[CoinFlipMove]) {
194        self.accepted += 1;
195        if self.accepted + self.rejected != self.steps {
196            panic!("{} {} {}", self.steps, self.rejected, self.accepted)
197        }
198    }
199
200    /// # Only implemented for testcases
201    /// Default implementation would suffice
202    fn steps_rejected(&mut self, _steps: &[CoinFlipMove]) {
203        self.rejected += 1;
204        if self.accepted + self.rejected != self.steps {
205            panic!("{} {} {}", self.steps, self.rejected, self.accepted)
206        }
207    }
208}
