6/// # Traits for quantities that all Wang Landau simulations have
7/// * see also: `WangLandauHist`
8/// * this trait is for convenience, so that you do not have
9///   to write all the trait bounds of, e.g.,  `WangLandauHist`, if you are
10///   not using functionality, that requires it
11pub trait WangLandau
12{
13    /// get current value of log_f
14    fn log_f(&self) -> f64;
15
16    /// # returns currently set threshold for log_f
17    fn log_f_threshold(&self) -> f64;
18    
19    /// Try to set the threshold. 
20    /// * `log_f_threshold > 0.0` has to be true
21    /// * `log_f_threshold` has to be finite
22    fn set_log_f_threshold(&mut self, log_f_threshold: f64) -> Result<f64, WangLandauErrors>;
23    
24    /// # Checks wang landau threshold
25    /// * `log_f <= log_f_threshold`
26    #[inline(always)]
27    fn is_finished(&self) -> bool
28    {
29        self.log_f() <= self.log_f_threshold()
30    }
31    
32    /// # Current (non normalized) estimate of ln(P(E))
33    /// * i.e., of the natural logarithm of the 
34    ///   probability density function
35    ///   for the requested interval
36    /// * this is what we are doing the simulations for
37    fn log_density(&self) -> &Vec<f64>;
38
39    /// # Current (non normalized) estimate of log10(P(E))
40    /// * i.e., of logarithm with base 10 of the 
41    ///   probability density function
42    ///   for the requested interval
43    /// * this is what we are doing the simulations for
44    fn log_density_base10(&self) -> Vec<f64>{
45        self.log_density()
46            .iter()
47            .map(|val| val * std::f64::consts::LOG10_E)
48            .collect()
49    }
50
51    /// # Current (non normalized) estimate of log_base(P(E))
52    /// * i.e., of logarithm with arbitrary base of the 
53    ///   probability density function
54    ///   for the requested interval
55    /// * this is what we are doing the simulations for
56    fn log_density_base(&self, base: f64) -> Vec<f64>{
57        let factor = std::f64::consts::E.log(base);
58        self.log_density()
59            .iter()
60            .map(|val| val * factor)
61            .collect()
62    }
63
64    /// Writes Information about the simulation to a file.
65    /// E.g. How many steps were performed.
66    fn write_log<W: Write>(&self, writer: W) -> Result<(), std::io::Error>;
67    
68    /// # Returns current wang landau mode
69    /// * see `WangLandauMode` for an explanation
70    fn mode(&self) -> WangLandauMode;
71    
72    /// # Counter
73    /// * how many wang Landau steps were performed until now?
74    /// * this does not include steps, that were performed to find a initial valid ensemble
75    fn step_counter(&self) -> usize;
76
77    /// # Counter
78    /// * how many wang Landau steps were performed until now?
79    /// * this includes steps, that were performed to find a initial valid ensemble
80    fn steps_total(&self) -> usize{
81        self.total_steps_accepted() + self.total_steps_rejected()
82    }
83
84    /// # How many steps were accepted until now?
85    /// * this includes steps, that were performed to find a initial valid ensemble
86    fn total_steps_accepted(&self) -> usize;
87
88    /// # How many steps were rejected until now?
89    /// * this includes steps, that were performed to find a initial valid ensemble
90    fn total_steps_rejected(&self) -> usize;
91
92    /// # Calculate, which fraction of steps were accepted
93    /// * this includes steps, that were performed to find a initial valid ensemble
94    /// * if no steps were performed, it returns `f64::NAN`
95    fn fraction_accepted_total(&self) -> f64 {
96        let total_acc = self.total_steps_accepted();
97        let total_steps = total_acc + self.total_steps_rejected();
98
99        if total_steps == 0 {
100            f64::NAN
101        } else {
102            total_acc as f64 / total_steps as f64
103        }
104    }
105
106    /// # Calculate, which fraction of steps were rejected
107    /// * this includes steps, that were performed to find a initial valid ensemble
108    /// * if no steps were performed, it returns `f64::NAN`
109    fn fraction_rejected_total(&self) -> f64 {
110        let total_rej = self.total_steps_rejected();
111        let total_steps = total_rej + self.total_steps_accepted();
112
113        if total_steps == 0 {
114            f64::NAN
115        } else {
116            total_rej as f64 / total_steps as f64
117        }
118    }
119}

122/// # trait to request a reference to the current (state of the) ensemble 
123/// * See also [WangLandauEEH](trait.WangLandauEEH.html)
124pub trait WangLandauEnsemble<E> : WangLandau
125{
126    /// return reference to current state of ensemble
127    fn ensemble(&self) -> &E;
128
129    /// # mutable reference to current state
130    /// * Intended for usecases where mutable access allows for a much more 
131    ///   efficient calculation of your calculations
132    /// # Safety
133    /// *you should not make any changes, that effect the 'energy' state of the system.
134    /// otherwise the WangLandau simulations will give false results!
135    /// * Note that I only use the unsafe keyword to force the user to acknowledge the above
136    unsafe fn ensemble_mut(&mut self) -> &mut E;
137}

139/// # trait to request the current histogram from a WangLandau simulation
140/// * Note: The histogram will likely be reset multiple times during a simulation
141/// * See also [WangLandauEEH](trait.WangLandauEEH.html)
142pub trait WangLandauHist<Hist> : WangLandau
143{
144    /// # returns current histogram
145    /// * **Note**: histogram will be reset multiple times during the simulation
146    /// * please refer to the [papers](struct.WangLandauAdaptive.html#adaptive-wanglandau-1t)
147    fn hist(&self) -> &Hist;
148}

150/// # trait to request the current energy from a WangLandau simulation
151/// * `None` if the energy was not calculated yet
152/// * See also [WangLandauEEH](trait.WangLandauEEH.html)
153pub trait WangLandauEnergy<Energy> : WangLandau
154{
155    /// returns the last accepted `Energy` calculated
156    /// `None` if no energy was calculated yet
157    fn energy(&self) -> Option<&Energy>;
158}

171pub(crate) trait WangLandau1TCalc<Hist> : WangLandauHist<Hist>
172where Hist: Histogram{
173    #[inline(always)]
174    fn log_f_1_t(&self) -> f64 
175    {
176        self.hist().bin_count() as f64 / self.step_counter() as f64
177    }
178}
