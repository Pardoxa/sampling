8/// # Implements histogram
9/// * anything that implements `Histogram` should also implement the trait `HistogramVal`
10pub trait Histogram {
11    /// # `self.hist[index] += 1`, `Err()` if `index` out of bounds
12    #[inline(always)]
13    fn increment_index(&mut self, index: usize) -> Result<(), HistErrors> {
14        self.increment_index_by(index, 1)
15    }
16
17    /// # `self.hist[index] += count`, `Err()` if `index` out of bounds
18    fn increment_index_by(&mut self, index: usize, count: usize) -> Result<(), HistErrors>;
19
20    /// # the created histogram
21    fn hist(&self) -> &Vec<usize>;
22    /// # How many bins the histogram contains
23    #[inline(always)]
24    fn bin_count(&self) -> usize {
25        self.hist().len()
26    }
27    /// reset the histogram to zero
28    fn reset(&mut self);
29
30    /// check if any bin was not hit yet
31    fn any_bin_zero(&self) -> bool {
32        self.hist().contains(&0)
33    }
34}

36/// * trait used for mapping values of arbitrary type `T` to bins
37/// * used to create a histogram
38pub trait HistogramVal<T> {
39    /// convert val to the respective histogram index
40    fn get_bin_index<V: Borrow<T>>(&self, val: V) -> Result<usize, HistErrors>;
41
42    /// count val. `Ok(index)`, if inside of hist, `Err(_)` if val is invalid
43    fn count_val<V: Borrow<T>>(&mut self, val: V) -> Result<usize, HistErrors>;
44
45    /// # binning borders
46    /// * the borders used to bin the values
47    fn bin_enum_iter(&'_ self) -> Box<dyn Iterator<Item = Bin<T>> + '_>;
48
49    /// does a value correspond to a valid bin?
50    fn is_inside<V: Borrow<T>>(&self, val: V) -> bool;
51
52    /// opposite of `is_inside`
53    fn not_inside<V: Borrow<T>>(&self, val: V) -> bool;
54
55    /// get the left most border (inclusive)
56    fn first_border(&self) -> T;
57
58    /// # get last border from the right
59    /// * Note: this border might be inclusive or exclusive
60    /// * check `last_border_is_inclusive` for finding it out
61    fn last_border(&self) -> T;
62
63    /// # True if last border is inclusive, false otherwise
64    /// * For most usecases this will return a constant value,
65    ///   as this is likely only dependent on the underlying type and not
66    ///   on something that changes dynamically
67    fn last_border_is_inclusive(&self) -> bool;
68
69    /// # calculates some sort of absolute distance to the nearest valid bin
70    /// * any invalid numbers (like NAN or INFINITY) should have the highest distance possible
71    /// * if a value corresponds to a valid bin, the distance should be zero
72    fn distance<V: Borrow<T>>(&self, val: V) -> f64;
73}

85/// # Your Interval is to large to sample in a reasonable amount of time? No problem
86/// In WangLandau or EntropicSampling, you can split your interval
87/// in smaller, overlapping intervals and "glue" them together later on
88pub trait HistogramPartition: Sized {
89    /// # partition the interval
90    /// * returns Vector of `n` histograms, that together cover the entire range of the original histogram
91    /// ## parameter
92    /// * `n` number of resulting intervals
93    /// * `overlap` How much overlap should there be?
94    /// ## To understand `overlap`, we have to look at the formula for the i_th interval in the result vector:
95    /// let `left` be the left border of `self` and `right` be the right border of self
96    /// * left border of interval i = left + i * (right - left) / (n + overlap)
97    /// * right border of interval i = left + (i + overlap) * (right - left) / (n + overlap)
98    fn overlapping_partition(
99        &self,
100        n: NonZeroUsize,
101        overlap: usize,
102    ) -> Result<Vec<Self>, HistErrors>;
103}

129/// Possible Errors of the traits `Histogram` and `HistogramVal`
130#[derive(Debug, Clone, Eq, PartialEq)]
131#[cfg_attr(feature = "serde_support", derive(Serialize, Deserialize))]
132pub enum HistErrors {
133    /// A histogram without any bins does not make sense!
134    NoBins,
135
136    /// Nothing can hit the bin! (left >= right?)
137    IntervalWidthZero,
138
139    /// Invalid value
140    OutsideHist,
141
142    /// Underflow occurred
143    Underflow,
144
145    /// Overflow occurred,
146    Overflow,
147
148    /// Error while casting to usize
149    UsizeCastError,
150
151    /// Something went wrong wile casting!
152    CastError,
153
154    /// Could be NAN, INFINITY or similar
155    InvalidVal,
156
157    /// Cannot create requested interval with
158    /// bins, that all have the same width!
159    ModuloError,
160
161    /// Cannot create the requested overlap!
162    ///
163    /// This can, for example, happen, if you
164    /// have the interval \[1,3\] and try to create more than two overlapping
165    /// intervals with overlap 1 from this, since we can only create \[1,2\] and  \[2,3\],
166    /// so only two intervals.
167    InvalidOverlap,
168
169    /// Unable to perform operation on empty slice
170    EmptySlice,
171
172    /// Cannot create the histograms due to Alignment issues
173    ///
174    /// Example:
175    ///
176    /// Lets say we have hist one with bins `[[0,1], [2,3], [4,5]]`
177    /// And another one with bins `[[5,6], [7,8]]`. Now we cannot create
178    /// an encapsulating histogram, since the number 5 appears in 2 different bins!
179    Alignment,
180}

182/// # Implements histogram
183/// * anything that implements `Histogram` should also implement the trait `HistogramVal`
184pub trait AtomicHistogram {
185    /// # `self.hist[index] += 1`, `Err()` if `index` out of bounds
186    #[inline(always)]
187    fn count_index(&self, index: usize) -> Result<(), HistErrors> {
188        self.count_multiple_index(index, 1)
189    }
190
191    /// # `self.hist[index] += count`, `Err()` if `index` out of bounds
192    fn count_multiple_index(&self, index: usize, count: usize) -> Result<(), HistErrors>;
193
194    /// # the created histogram
195    /// Since this uses atomics, you can also write to the underlying hist yourself,
196    /// if you so desire
197    fn hist(&self) -> &[AtomicUsize];
198
199    /// # How many bins the histogram contains
200    #[inline(always)]
201    fn bin_count(&self) -> usize {
202        self.hist().len()
203    }
204    /// reset the histogram to zero
205    fn reset(&mut self);
206
207    /// # check if any bin was not hit yet
208    /// * Uses SeqCst Ordering
209    fn any_bin_zero(&self) -> bool {
210        self.hist()
211            .iter()
212            .any(|val| val.load(std::sync::atomic::Ordering::SeqCst) == 0)
213    }
214}
