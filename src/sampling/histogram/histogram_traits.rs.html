13/// # Implements histogram
14/// * anything that implements `Histogram` should also implement the trait `HistogramVal`
15pub trait Histogram {
16    /// # `self.hist[index] += 1`, `Err()` if `index` out of bounds
17    #[inline(always)]
18    fn increment_index(&mut self, index: usize) -> Result<(), HistErrors>{
19        self.increment_index_by(index, 1)
20    }
21
22    /// # `self.hist[index] += count`, `Err()` if `index` out of bounds
23    fn increment_index_by(&mut self, index: usize, count: usize) -> Result<(), HistErrors>;
24
25    /// # the created histogram
26    fn hist(&self) -> &Vec<usize>;
27    /// # How many bins the histogram contains
28    #[inline(always)]
29    fn bin_count(&self) -> usize
30    {
31        self.hist().len()
32    }
33    /// reset the histogram to zero
34    fn reset(&mut self);
35
36    /// check if any bin was not hit yet
37    fn any_bin_zero(&self) -> bool
38    {
39        self.hist()
40            .iter()
41            .any(|&val| val == 0)
42    }
43}

47/// * trait used for mapping values of arbitrary type `T` to bins
48/// * used to create a histogram
49pub trait HistogramVal<T>{
50    /// convert val to the respective histogram index
51    fn get_bin_index<V: Borrow<T>>(&self, val: V) -> Result<usize, HistErrors>;
52    
53    /// count val. `Ok(index)`, if inside of hist, `Err(_)` if val is invalid
54    fn count_val<V: Borrow<T>>(&mut self, val: V) -> Result<usize, HistErrors>;
55    
56    /// # binning borders
57    /// * the borders used to bin the values
58    fn bin_enum_iter(&'_ self) -> Box<dyn Iterator<Item=Bin<T>> + '_>;
59    
60    /// does a value correspond to a valid bin?
61    fn is_inside<V: Borrow<T>>(&self, val: V) -> bool;
62    
63    /// opposite of `is_inside`
64    fn not_inside<V: Borrow<T>>(&self, val: V) -> bool;
65    
66    /// get the left most border (inclusive)
67    fn first_border(&self) -> T;
68
69    /// # get last border from the right
70    /// * Note: this border might be inclusive or exclusive
71    /// * check `last_border_is_inclusive` for finding it out
72    fn last_border(&self) -> T;
73
74    /// # True if last border is inclusive, false otherwise
75    /// * For most usecases this will return a constant value,
76    ///   as this is likely only dependent on the underlying type and not 
77    ///   on something that changes dynamically
78    fn last_border_is_inclusive(&self) -> bool;
79
80    /// # calculates some sort of absolute distance to the nearest valid bin
81    /// * any invalid numbers (like NAN or INFINITY) should have the highest distance possible
82    /// * if a value corresponds to a valid bin, the distance should be zero
83    fn distance<V: Borrow<T>>(&self, val: V) -> f64;
84}

100/// # Your Interval is to large to sample in a reasonable amount of time? No problem
101/// In WangLandau or EntropicSampling, you can split your interval
102/// in smaller, overlapping intervals and "glue" them together later on
103pub trait HistogramPartition: Sized
104{
105    /// # partition the interval
106    /// * returns Vector of `n` histograms, that together cover the entire range of the original histogram
107    /// ## parameter
108    /// * `n` number of resulting intervals
109    /// * `overlap` How much overlap should there be?
110    /// ## To understand `overlap`, we have to look at the formula for the i_th interval in the result vector:
111    /// let `left` be the left border of `self` and `right` be the right border of self
112    /// * left border of interval i = left + i * (right - left) / (n + overlap)
113    /// * right border of interval i = left + (i + overlap) * (right - left) / (n + overlap)
114    fn overlapping_partition(&self, n: NonZeroUsize, overlap: usize) -> Result<Vec<Self>, HistErrors>;
115}

141/// Possible Errors of the traits `Histogram` and `HistogramVal`
142#[derive(Debug, Clone, Eq, PartialEq)]
143#[cfg_attr(feature = "serde_support", derive(Serialize, Deserialize))]
144pub enum HistErrors{
145    /// A histogram without any bins does not make sense!
146    NoBins,
147
148    /// Nothing can hit the bin! (left >= right?)
149    IntervalWidthZero,
150
151    /// Invalid value
152    OutsideHist,
153
154    /// Underflow occurred
155    Underflow,
156
157    /// Overflow occurred,
158    Overflow,
159
160    /// Error while casting to usize
161    UsizeCastError,
162    
163    /// Something went wrong wile casting!
164    CastError,
165
166    /// Could be NAN, INFINITY or similar
167    InvalidVal,
168
169    /// Cannot create requested interval with 
170    /// bins, that all have the same width!
171    ModuloError,
172
173    /// Cannot create the requested overlap!
174    /// 
175    /// This can, for example, happen, if you 
176    /// have the interval \[1,3\] and try to create more than two overlapping 
177    /// intervals with overlap 1 from this, since we can only create \[1,2\] and  \[2,3\],
178    /// so only two intervals.
179    InvalidOverlap,
180
181    /// Unable to perform operation on empty slice
182    EmptySlice,
183
184    /// Cannot create the histograms due to Alignment issues
185    /// 
186    /// Example:
187    /// 
188    /// Lets say we have hist one with bins `[[0,1], [2,3], [4,5]]`
189    /// And another one with bins `[[5,6], [7,8]]`. Now we cannot create 
190    /// an encapsulating histogram, since the number 5 appears in 2 different bins!
191    Alignment
192}

194/// # Implements histogram
195/// * anything that implements `Histogram` should also implement the trait `HistogramVal`
196pub trait AtomicHistogram {
197    /// # `self.hist[index] += 1`, `Err()` if `index` out of bounds
198    #[inline(always)]
199    fn count_index(&self, index: usize) -> Result<(), HistErrors>{
200        self.count_multiple_index(index, 1)
201    }
202
203    /// # `self.hist[index] += count`, `Err()` if `index` out of bounds
204    fn count_multiple_index(&self, index: usize, count: usize) -> Result<(), HistErrors>;
205
206    /// # the created histogram
207    /// Since this uses atomics, you can also write to the underlying hist yourself,
208    /// if you so desire
209    fn hist(&self) -> &[AtomicUsize];
210
211    /// # How many bins the histogram contains
212    #[inline(always)]
213    fn bin_count(&self) -> usize
214    {
215        self.hist().len()
216    }
217    /// reset the histogram to zero
218    fn reset(&mut self);
219
220    /// # check if any bin was not hit yet
221    /// * Uses SeqCst Ordering
222    fn any_bin_zero(&self) -> bool
223    {
224        self.hist()
225            .iter()
226            .any(|val| val.load(std::sync::atomic::Ordering::SeqCst) == 0)
227    }
228}
