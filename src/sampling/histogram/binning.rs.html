12/// # Definition of a Bin
13/// * Note: Most (currently all) implementations use more efficient representations of the bins underneath,
14///   but are capable of returning the bins in this representation on request
15#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
16#[cfg_attr(feature = "serde_support", derive(Serialize, Deserialize))]
17pub enum Bin<T> {
18    /// The bin consists of a single value. A value is inside the bin if it equals this value
19    SingleValued(T),
20    /// The bin is defined by two inclusive borders (left, right).
21    /// a value is inside the bin, if left <= value <= right
22    InclusiveInclusive(T, T),
23    /// The bin is defined by an inclusive and an exclusive border (left, right).
24    /// a value is inside the bin, if left <= value < right
25    InclusiveExclusive(T, T),
26    /// The bin is defined by an exclusive and an inclusive border (left, right).
27    /// a value is inside the bin, if left < value <= right
28    ExclusiveInclusive(T, T),
29    /// The bin is defined by two exclusive borders (left, right).
30    /// a value is inside the bin, if left < value < right
31    ExclusiveExclusive(T, T),
32}

34/// # Implements Binning
35/// Part of a histogram, but without the capability of counting stuff, i.e.,
36/// you can use this to iterate through the bins or to get the bin index that a certain value would
37/// correspond to, but it does not contain counters to track this.
38///
39/// You can use this to create histograms, see [GenericHist] or [AtomicGenericHist]
40pub trait Binning<T> {
41    /// convert val to the respective binning index
42    fn get_bin_index<V: Borrow<T>>(&self, val: V) -> Option<usize>;
43
44    /// # Get the number of underlying bins
45    /// * note: if more than usize::MAX bins are there, usize::MAX is returned
46    fn get_bin_len(&self) -> usize;
47
48    /// # Iterates over all bins
49    /// Note: Most (currently all) implementations use more efficient representations of the bins underneath,
50    ///     but are capable of returning the bins in this representation on request.
51    ///     So it's better to use the native iterator instead, if you can.
52    fn bin_iter(&self) -> Box<dyn Iterator<Item = Bin<T>>>;
53
54    /// Does a value correspond to a valid bin?
55    fn is_inside<V: Borrow<T>>(&self, val: V) -> bool;
56
57    /// Opposite of `is_inside`, so: is the value outside the valid range?
58    fn not_inside<V: Borrow<T>>(&self, val: V) -> bool;
59
60    /// get the left most border (inclusive). This is the first value that lies inside the binning
61    fn first_border(&self) -> T;
62
63    /// * If the last border is inclusive, this returns the largest value that is still inside the binning.
64    /// * If the last border is exclusive, this is the first value which is **not** inside the binning.
65    fn last_border(&self) -> T;
66
67    /// # True if last border is inclusive, false otherwise
68    /// For most use cases this will return a constant value,
69    /// as this is likely only dependent on the underlying type and not
70    /// on something that changes dynamically
71    fn last_border_is_inclusive(&self) -> bool;
72
73    /// # calculates some sort of absolute distance to the nearest valid bin
74    /// * any invalid numbers (like NAN or INFINITY) should have the highest distance possible
75    /// * if a value corresponds to a valid bin, the distance should be zero
76    fn distance<V: Borrow<T>>(&self, val: V) -> f64;
77
78    /// # Convert binning into [GenericHist]
79    /// Useful histogram for single threaded context's.
80    /// Otherwise [AtomicGenericHist] might be more useful (see also [Binning::to_generic_atomic_hist])
81    fn to_generic_hist(self) -> GenericHist<Self, T>
82    where
83        Self: Sized,
84    {
85        GenericHist::new(self)
86    }
87
88    /// # Convert binning into a [AtomicGenericHist]
89    /// Useful histogram if you want to create the histogram in parallel, but otherwise has less functionality
90    /// than [GenericHist] (see also [Binning::to_generic_hist])
91    fn to_generic_atomic_hist(self) -> AtomicGenericHist<Self, T>
92    where
93        Self: Sized,
94    {
95        AtomicGenericHist::new(self)
96    }
97}

99/// # Trait used to display bins
100/// * This is, e.g., used by the glue writers to write the bins of the merged results
101pub trait BinDisplay {
102    /// What type does the displayable BinEntry have?
103    type BinEntry;
104
105    /// # Iterator over all the bins
106    /// * you might require to use this if you are working with generics
107    /// * if you are working with a specific type there is usually a more efficient implementation
108    ///   that did not require the usage of dynamic traits (`dyn`) and that are thus more efficient,
109    ///   consider using those instead
110    fn display_bin_iter(&'_ self) -> Box<dyn Iterator<Item = Self::BinEntry> + '_>;
111
112    /// # For writing a bin
113    /// * How to write a bin to a file? If a bin consists, e.g., of an exclusive and inclusive border this
114    ///   might require the writing of two values. It could also be a single value instead.
115    ///   It should be something the user expects from your binning, see write header
116    fn write_bin<W: std::io::Write>(entry: &Self::BinEntry, writer: W) -> std::io::Result<()>;
117
118    /// # Writing the header of the bin
119    /// * This is intended to name, e.g., a column in a file. Output could be "SingleBin" or "BinBorderExclusive BinBorderInclusive"
120    ///   and so on
121    fn write_header<W: std::io::Write>(&self, writer: W) -> std::io::Result<()>;
122}
