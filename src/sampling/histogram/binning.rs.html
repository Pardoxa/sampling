13/// # Definition of a Bin
14/// * Note: Most (currently all) implementations use more efficient representations of the bins underneath,
15///   but are capable of returning the bins in this representation on request
16#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
17#[cfg_attr(feature = "serde_support", derive(Serialize, Deserialize))]
18pub enum Bin<T>
19{
20    /// The bin consists of a single value. A value is inside the bin if it equals this value
21    SingleValued(T),
22    /// The bin is defined by two inclusive borders (left, right).
23    /// a value is inside the bin, if left <= value <= right
24    InclusiveInclusive(T, T),
25    /// The bin is defined by an inclusive and an exclusive border (left, right).
26    /// a value is inside the bin, if left <= value < right
27    InclusiveExclusive(T, T),
28    /// The bin is defined by an exclusive and an inclusive border (left, right).
29    /// a value is inside the bin, if left < value <= right
30    ExclusiveInclusive(T, T),
31    /// The bin is defined by two exclusive borders (left, right).
32    /// a value is inside the bin, if left < value < right
33    ExclusiveExclusive(T, T)
34}

37/// # Implements Binning
38/// Part of a histogram, but without the capability of counting stuff, i.e.,
39/// you can use this to iterate through the bins or to get the bin index that a certain value would 
40/// correspond to, but it does not contain counters to track this.
41/// 
42/// You can use this to create histograms, see [GenericHist] or [AtomicGenericHist]
43pub trait Binning<T>{
44    /// convert val to the respective binning index
45    fn get_bin_index<V: Borrow<T>>(&self, val: V) -> Option<usize>;
46
47    /// # Get the number of underlying bins
48    /// * note: if more than usize::MAX bins are there, usize::MAX is returned
49    fn get_bin_len(&self) -> usize;
50
51    /// # Iterates over all bins
52    /// Note: Most (currently all) implementations use more efficient representations of the bins underneath,
53    ///     but are capable of returning the bins in this representation on request. 
54    ///     So it's better to use the native iterator instead, if you can.
55    fn bin_iter(&self) -> Box<dyn Iterator<Item=Bin<T>>>;
56
57    /// Does a value correspond to a valid bin?
58    fn is_inside<V: Borrow<T>>(&self, val: V) -> bool;
59
60    /// Opposite of `is_inside`, so: is the value outside the valid range?
61    fn not_inside<V: Borrow<T>>(&self, val: V) -> bool;
62
63    /// get the left most border (inclusive). This is the first value that lies inside the binning
64    fn first_border(&self) -> T;
65
66    /// * If the last border is inclusive, this returns the largest value that is still inside the binning.
67    /// * If the last border is exclusive, this is the first value which is **not** inside the binning.
68    fn last_border(&self) -> T;
69
70    /// # True if last border is inclusive, false otherwise
71    /// For most use cases this will return a constant value,
72    /// as this is likely only dependent on the underlying type and not 
73    /// on something that changes dynamically
74    fn last_border_is_inclusive(&self) -> bool;
75
76    /// # calculates some sort of absolute distance to the nearest valid bin
77    /// * any invalid numbers (like NAN or INFINITY) should have the highest distance possible
78    /// * if a value corresponds to a valid bin, the distance should be zero
79    fn distance<V: Borrow<T>>(&self, val: V) -> f64;
80
81    /// # Convert binning into [GenericHist]
82    /// Useful histogram for single threaded context's. 
83    /// Otherwise [AtomicGenericHist] might be more useful (see also [Binning::to_generic_atomic_hist])
84    fn to_generic_hist(self) -> GenericHist<Self, T>
85    where Self: Sized
86    {
87        GenericHist::new(self)
88    }
89
90    /// # Convert binning into a [AtomicGenericHist]
91    /// Useful histogram if you want to create the histogram in parallel, but otherwise has less functionality 
92    /// than [GenericHist] (see also [Binning::to_generic_hist])
93    fn to_generic_atomic_hist(self) -> AtomicGenericHist<Self, T>
94    where Self: Sized
95    {
96        AtomicGenericHist::new(self)
97    }
98}

100/// # Trait used to display bins
101/// * This is, e.g., used by the glue writers to write the bins of the merged results
102pub trait BinDisplay {
103    /// What type does the displayable BinEntry have?
104    type BinEntry;
105
106    /// # Iterator over all the bins
107    /// * you might require to use this if you are working with generics
108    /// * if you are working with a specific type there is usually a more efficient implementation
109    ///   that did not require the usage of dynamic traits (`dyn`) and that are thus more efficient,
110    ///   consider using those instead
111    fn display_bin_iter(&'_ self) -> Box<dyn Iterator<Item=Self::BinEntry> + '_>;
112
113    /// # For writing a bin
114    /// * How to write a bin to a file? If a bin consists, e.g., of an exclusive and inclusive border this 
115    ///   might require the writing of two values. It could also be a single value instead.
116    ///   It should be something the user expects from your binning, see write header
117    fn write_bin<W: std::io::Write>(entry: &Self::BinEntry, writer: W) -> std::io::Result<()>;
118
119    /// # Writing the header of the bin
120    /// * This is intended to name, e.g., a column in a file. Output could be "SingleBin" or "BinBorderExclusive BinBorderInclusive"
121    ///   and so on
122    fn write_header<W: std::io::Write>(&self, writer: W) -> std::io::Result<()>;
123}
