23/// Generic binning meant for any integer type
24#[derive(Debug, Clone, PartialEq, Eq)]
25#[cfg_attr(feature = "serde_support", derive(Serialize, Deserialize))]
26pub struct BinningWithWidth<T>
27where T: HasUnsignedVersion
28{
29    /// left bin border, inclusive
30    start: T,
31    /// right bin border, inclusive
32    end_inclusive: T,
33    /// how many numbers are in one bin?
34    bin_width: T,
35    /// bin width as unsigned
36    bin_width_unsigned: T::Unsigned
37}
38
39macro_rules! other_binning {
40    (
41        $t:ty
42    ) => {
43        
44        paste!{
45            #[doc = "Efficient binning for `" $t "` with arbitrary width"]
46            pub type [<Binning $t:upper>] = BinningWithWidth<$t>;
47        }
48        
49        impl paste!{[<Binning $t:upper>]}{
50            /// # Create a new Binning
51            /// * both borders are inclusive
52            /// * each bin has width bin_width
53            /// # Panics
54            /// * if `start` is smaller than `end_inclusive`
55            /// * if bin_width <= 0
56            /// # Err
57            /// Result is Err if start and end are mismatched with the bin_width, i.e., 
58            /// it is impossible to create the binning due to the integer nature of our types
59            #[inline(always)]
60            pub fn new_inclusive(start: $t, end_inclusive: $t, bin_width: $t) -> Result<Self, <$t as HasUnsignedVersion>::Unsigned>{
61                assert!(start <= end_inclusive);
62                assert!(bin_width > 0);
63                let bin_width_unsigned = bin_width as <$t as HasUnsignedVersion>::Unsigned;
64                let this = Self{
65                    start, 
66                    end_inclusive,
67                    bin_width,
68                    bin_width_unsigned
69                };
70
71                // Check if the requested bin_width makes sense
72                let left = to_u(start);
73                let right = to_u(end_inclusive);
74                let res = ((right - left) % bin_width_unsigned + 1) % bin_width_unsigned;
75                // now res needs to be 0 for this to be a valid config
76                if res != 0{
77                    Err(res)
78                } else {
79                    Ok(this)
80                }
81            }
82
83            /// Get left border, inclusive
84            #[inline(always)]
85            pub const fn left(&self) -> $t {
86                self.start
87            }
88
89            /// Get right border, inclusive
90            #[inline(always)]
91            pub const fn right(&self) -> $t
92            {
93                self.end_inclusive
94            }
95
96            paste!{
97                #[doc="# Returns the range covered by the bins as a `RangeInclusive<" $t ">`"]           
98                #[inline(always)]
99                pub const fn range_inclusive(&self) -> RangeInclusive<$t>
100                {
101                    self.start..=self.end_inclusive
102                }
103            }
104
105            paste!{
106                #[doc = "# Iterator over all the bins\n\
107                Here the bins are represented as `RangeInclusive<" $t ">` \n\
108                # Example: \
109                \n\
110                ```
111use sampling::histogram::" [<Binning $t:upper>] ";\n\
112                let binning = " [<Binning $t:upper>] "::new_inclusive(2,7,2).unwrap();\n\
113                let vec: Vec<_> = binning.native_bin_iter().collect();\n\
114                assert_eq!(&vec, &[(2..=3), (4..=5), (6..=7)]);\n\
115                ```"]
116                #[inline]
117                pub fn native_bin_iter(&self) -> impl Iterator<Item=RangeInclusive<$t>>
118                {
119                    let width = self.bin_width;
120                    BinModIterHelper::new_unchecked(self.start, self.end_inclusive, width)
121                }
122            }
123
124            /// # The amount of bins -1
125            /// * minus 1 because if the bins are of width 1 and are going over the entire range of the type,
126            ///   then we cannot represent the number of bins as this type
127            /// 
128            /// # Example
129            /// If we look at an u8 and the range from 0 to 255, then this is 256 bins, which 
130            /// cannot be represented as u8. To combat this, I return bins - 1.
131            /// This works, because we always have at least 1 bin
132            #[inline(always)]
133            pub fn bins_m1(&self) -> <$t as HasUnsignedVersion>::Unsigned{
134                let left = to_u(self.start);
135                let right = to_u(self.end_inclusive);
136
137                (right - left) / self.bin_width_unsigned
138            }
139
140            /// # Get the respective bin in native unsigned
141            #[inline(always)]
142            pub fn get_bin_index_native<V: Borrow<$t>>(&self, val: V) -> Option<<$t as HasUnsignedVersion>::Unsigned>{
143                let val = *val.borrow();
144                if self.is_inside(val)
145                {
146                    let index = (to_u(val) - to_u(self.start)) / self.bin_width_unsigned;
147                    Some(index)
148                } else {
149                    None
150                }
151            }
152        }
153

155        impl GenericHist<paste!{[<Binning $t:upper>]}, $t>{
156            /// # Iterate over bins and hits
157            /// Returns an iterator, which gives yields (bin, hits), i.e.,
158            /// a RangeInclusive that represents the bin
159            /// and the corresponding number of hits
160            pub fn bin_hits_iter(&'_ self) -> impl Iterator<Item=(RangeInclusive<$t>, usize)> + '_
161            {
162                self.binning()
163                    .native_bin_iter()
164                    .zip(self.hist().iter().copied())
165            }
166        }

170        impl Binning<$t> for paste!{[<Binning $t:upper>]} {
171            #[inline(always)]
172            fn get_bin_len(&self) -> usize 
173            {
174                (self.bins_m1() as usize).saturating_add(1)
175            }
176
177            /// # Get the respective bin index
178            /// * Note: Obviously this breaks when the bin index cannot be represented as 
179            ///   `usize`
180            #[inline(always)]
181            fn get_bin_index<V: Borrow<$t>>(&self, val: V) -> Option<usize>{
182                self.get_bin_index_native(val)
183                    .map(|v| v as usize)
184            }
185
186            /// Does a value correspond to a valid bin?
187            #[inline(always)]
188            fn is_inside<V: Borrow<$t>>(&self, val: V) -> bool{
189                (self.start..=self.end_inclusive).contains(val.borrow())
190            }
191
192            /// # Opposite of `is_inside`
193            /// * I could also have called this `is_outside`, but I didn't
194            #[inline(always)]
195            fn not_inside<V: Borrow<$t>>(&self, val: V) -> bool{
196                !self.is_inside(val)
197            }
198
199            /// get the left most border (inclusive)
200            fn first_border(&self) -> $t{
201                self.start
202            }
203
204            fn last_border(&self) -> $t{
205                self.end_inclusive
206            }
207
208            #[inline(always)]
209            fn last_border_is_inclusive(&self) -> bool
210            {
211                true
212            }
213
214            /// # calculates some sort of absolute distance to the nearest valid bin
215            /// * if a value corresponds to a valid bin, the distance is zero
216            fn distance<V: Borrow<$t>>(&self, v: V) -> f64{
217                let val = v.borrow();
218                if self.is_inside(val){
219                    0.0
220                } else {
221                    let dist = if *val < self.start {
222                        to_u(self.start) - to_u(*val)
223                    } else {
224                        to_u(*val) - to_u(self.end_inclusive)
225                    };
226                    dist as f64
227                }
228            }
229
230            /// # Iterates over all bins
231            /// * Note: This implementation uses a more efficient representations of the bins underneath,
232            ///   but is capable of returning the bins in this representation on request
233            /// * Note also that this `Binning`  implements another method for the bin borders, i.e., `native_bin_iter`.
234            ///   Consider using that instead, as it is more efficient
235            fn bin_iter(&self) -> Box<dyn Iterator<Item=Bin<$t>>>{
236                let iter = self
237                    .native_bin_iter()
238                    .map(
239                        |range| {
240                            let (start, end) = range.into_inner();
241                            Bin::InclusiveInclusive(start, end)
242                        }
243                    );
244                Box::new(iter)
245            }
246        }

248        impl HistogramCombine for GenericHist<paste!{[<Binning $t:upper>]}, $t>
249        {
250            fn align<S>(&self, right: S)-> Result<usize, super::HistErrors>
251                where S: Borrow<Self> {
252                let self_bins = self.binning();
253                let right_bins = right.borrow().binning();
254                
255                if self_bins.bin_width != right_bins.bin_width{
256                    return Err(HistErrors::ModuloError);
257                }
258            
259                let right_first_border = right_bins.first_border();
260            
261                let idx = self_bins.get_bin_index(right_first_border)
262                    .ok_or(HistErrors::OutsideHist)?;
263            
264                // now we have the index, but we need to check the alignment!
265                let distance = to_u(right_first_border) - to_u(self_bins.first_border());
266                let modulo = distance % self_bins.bin_width_unsigned;
267                if modulo != 0 {
268                    return Err(HistErrors::Alignment);
269                }
270                Ok(idx)
271            }
272        
273            fn encapsulating_hist<S>(hists: &[S]) -> Result<Self, super::HistErrors>
274                where S: Borrow<Self> {
275                if hists.is_empty(){
276                    return Err(HistErrors::EmptySlice);
277                }
278                let first = hists[0].borrow().binning();
279                let width = first.bin_width;
280                let mut left = first.first_border();
281                let mut right = first.last_border();
282                for other in hists[1..].iter(){
283                    let binning = other.borrow().binning();
284                    if width != binning.bin_width{
285                        return Err(HistErrors::ModuloError);
286                    }
287                    left = left.min(binning.first_border());
288                    right = right.max(binning.last_border());
289                }
290                // now I first create the binning, then I check if all the intervals aligned properly
291                let new_binning = <paste!{[<Binning $t:upper>]}>::new_inclusive(
292                    left, 
293                    right, 
294                    width
295                ).map_err(|_| HistErrors::ModuloError)?;
296            
297                let new_first_border = to_u(new_binning.first_border());
298                let bin_width_u = first.bin_width_unsigned;
299            
300                for hist in hists.iter(){
301                    let binning = hist.borrow().binning();
302                    let distance = to_u(binning.first_border()) - new_first_border;
303                    let modulo = distance % bin_width_u;
304                    if modulo != 0 {
305                        return Err(HistErrors::Alignment);
306                    }
307                }
308                Ok(GenericHist::new(new_binning))
309            }
310        }
311    };

349        let vec: Vec<_> = binning.native_bin_iter().collect();
