12/// Generic binning meant for any integer type
13#[derive(Debug, Clone, PartialEq, Eq)]
14#[cfg_attr(feature = "serde_support", derive(Serialize, Deserialize))]
15pub struct BinningWithWidth<T>
16where
17    T: HasUnsignedVersion,
18{
19    /// left bin border, inclusive
20    start: T,
21    /// right bin border, inclusive
22    end_inclusive: T,
23    /// how many numbers are in one bin?
24    bin_width: T,
25    /// bin width as unsigned
26    bin_width_unsigned: T::Unsigned,
27}

29macro_rules! other_binning {
30    (
31        $t:ty
32    ) => {

34        paste!{
35            #[doc = "Efficient binning for `" $t "` with arbitrary width"]
36            pub type [<Binning $t:upper>] = BinningWithWidth<$t>;
37        }
38
39        impl paste!{[<Binning $t:upper>]}{
40            /// # Create a new Binning
41            /// * both borders are inclusive
42            /// * each bin has width bin_width
43            /// # Panics
44            /// * if `start` is smaller than `end_inclusive`
45            /// * if bin_width <= 0
46            /// # Err
47            /// Result is Err if start and end are mismatched with the bin_width, i.e.,
48            /// it is impossible to create the binning due to the integer nature of our types
49            #[inline(always)]
50            pub fn new_inclusive(start: $t, end_inclusive: $t, bin_width: $t) -> Result<Self, <$t as HasUnsignedVersion>::Unsigned>{
51                assert!(start <= end_inclusive);
52                assert!(bin_width > 0);
53                let bin_width_unsigned = bin_width as <$t as HasUnsignedVersion>::Unsigned;
54                let this = Self{
55                    start,
56                    end_inclusive,
57                    bin_width,
58                    bin_width_unsigned
59                };
60
61                // Check if the requested bin_width makes sense
62                let left = to_u(start);
63                let right = to_u(end_inclusive);
64                let res = ((right - left) % bin_width_unsigned + 1) % bin_width_unsigned;
65                // now res needs to be 0 for this to be a valid config
66                if res != 0{
67                    Err(res)
68                } else {
69                    Ok(this)
70                }
71            }
72
73            /// Get left border, inclusive
74            #[inline(always)]
75            pub const fn left(&self) -> $t {
76                self.start
77            }
78
79            /// Get right border, inclusive
80            #[inline(always)]
81            pub const fn right(&self) -> $t
82            {
83                self.end_inclusive
84            }
85
86            paste!{
87                #[doc="# Returns the range covered by the bins as a `RangeInclusive<" $t ">`"]
88                #[inline(always)]
89                pub const fn range_inclusive(&self) -> RangeInclusive<$t>
90                {
91                    self.start..=self.end_inclusive
92                }
93            }
94
95            paste!{
96                #[doc = "# Iterator over all the bins\n\
97                Here the bins are represented as `RangeInclusive<" $t ">` \n\
98                # Example: \
99                \n\
100                ```
101use sampling::histogram::" [<Binning $t:upper>] ";\n\
102                let binning = " [<Binning $t:upper>] "::new_inclusive(2,7,2).unwrap();\n\
103                let vec: Vec<_> = binning.native_bin_iter().collect();\n\
104                assert_eq!(&vec, &[(2..=3), (4..=5), (6..=7)]);\n\
105                ```"]
106                #[inline]
107                pub fn native_bin_iter(&self) -> impl Iterator<Item=RangeInclusive<$t>>
108                {
109                    let width = self.bin_width;
110                    BinModIterHelper::new_unchecked(self.start, self.end_inclusive, width)
111                }
112            }
113
114            /// # The amount of bins -1
115            /// * minus 1 because if the bins are of width 1 and are going over the entire range of the type,
116            ///   then we cannot represent the number of bins as this type
117            ///
118            /// # Example
119            /// If we look at an u8 and the range from 0 to 255, then this is 256 bins, which
120            /// cannot be represented as u8. To combat this, I return bins - 1.
121            /// This works, because we always have at least 1 bin
122            #[inline(always)]
123            pub fn bins_m1(&self) -> <$t as HasUnsignedVersion>::Unsigned{
124                let left = to_u(self.start);
125                let right = to_u(self.end_inclusive);
126
127                (right - left) / self.bin_width_unsigned
128            }
129
130            /// # Get the respective bin in native unsigned
131            #[inline(always)]
132            pub fn get_bin_index_native<V: Borrow<$t>>(&self, val: V) -> Option<<$t as HasUnsignedVersion>::Unsigned>{
133                let val = *val.borrow();
134                if self.is_inside(val)
135                {
136                    let index = (to_u(val) - to_u(self.start)) / self.bin_width_unsigned;
137                    Some(index)
138                } else {
139                    None
140                }
141            }
142        }

145        impl GenericHist<paste!{[<Binning $t:upper>]}, $t>{
146            /// # Iterate over bins and hits
147            /// Returns an iterator, which gives yields (bin, hits), i.e.,
148            /// a RangeInclusive that represents the bin
149            /// and the corresponding number of hits
150            pub fn bin_hits_iter(&'_ self) -> impl Iterator<Item=(RangeInclusive<$t>, usize)> + '_
151            {
152                self.binning()
153                    .native_bin_iter()
154                    .zip(self.hist().iter().copied())
155            }
156        }

160        impl Binning<$t> for paste!{[<Binning $t:upper>]} {
161            #[inline(always)]
162            fn get_bin_len(&self) -> usize
163            {
164                (self.bins_m1() as usize).saturating_add(1)
165            }
166
167            /// # Get the respective bin index
168            /// * Note: Obviously this breaks when the bin index cannot be represented as
169            ///   `usize`
170            #[inline(always)]
171            fn get_bin_index<V: Borrow<$t>>(&self, val: V) -> Option<usize>{
172                self.get_bin_index_native(val)
173                    .map(|v| v as usize)
174            }
175
176            /// Does a value correspond to a valid bin?
177            #[inline(always)]
178            fn is_inside<V: Borrow<$t>>(&self, val: V) -> bool{
179                (self.start..=self.end_inclusive).contains(val.borrow())
180            }
181
182            /// # Opposite of `is_inside`
183            /// * I could also have called this `is_outside`, but I didn't
184            #[inline(always)]
185            fn not_inside<V: Borrow<$t>>(&self, val: V) -> bool{
186                !self.is_inside(val)
187            }
188
189            /// get the left most border (inclusive)
190            fn first_border(&self) -> $t{
191                self.start
192            }
193
194            fn last_border(&self) -> $t{
195                self.end_inclusive
196            }
197
198            #[inline(always)]
199            fn last_border_is_inclusive(&self) -> bool
200            {
201                true
202            }
203
204            /// # calculates some sort of absolute distance to the nearest valid bin
205            /// * if a value corresponds to a valid bin, the distance is zero
206            fn distance<V: Borrow<$t>>(&self, v: V) -> f64{
207                let val = v.borrow();
208                if self.is_inside(val){
209                    0.0
210                } else {
211                    let dist = if *val < self.start {
212                        to_u(self.start) - to_u(*val)
213                    } else {
214                        to_u(*val) - to_u(self.end_inclusive)
215                    };
216                    dist as f64
217                }
218            }
219
220            /// # Iterates over all bins
221            /// * Note: This implementation uses a more efficient representations of the bins underneath,
222            ///   but is capable of returning the bins in this representation on request
223            /// * Note also that this `Binning`  implements another method for the bin borders, i.e., `native_bin_iter`.
224            ///   Consider using that instead, as it is more efficient
225            fn bin_iter(&self) -> Box<dyn Iterator<Item=Bin<$t>>>{
226                let iter = self
227                    .native_bin_iter()
228                    .map(
229                        |range| {
230                            let (start, end) = range.into_inner();
231                            Bin::InclusiveInclusive(start, end)
232                        }
233                    );
234                Box::new(iter)
235            }
236        }

238        impl HistogramCombine for GenericHist<paste!{[<Binning $t:upper>]}, $t>
239        {
240            fn align<S>(&self, right: S)-> Result<usize, super::HistErrors>
241                where S: Borrow<Self> {
242                let self_bins = self.binning();
243                let right_bins = right.borrow().binning();
244
245                if self_bins.bin_width != right_bins.bin_width{
246                    return Err(HistErrors::ModuloError);
247                }
248
249                let right_first_border = right_bins.first_border();
250
251                let idx = self_bins.get_bin_index(right_first_border)
252                    .ok_or(HistErrors::OutsideHist)?;
253
254                // now we have the index, but we need to check the alignment!
255                let distance = to_u(right_first_border) - to_u(self_bins.first_border());
256                let modulo = distance % self_bins.bin_width_unsigned;
257                if modulo != 0 {
258                    return Err(HistErrors::Alignment);
259                }
260                Ok(idx)
261            }
262
263            fn encapsulating_hist<S>(hists: &[S]) -> Result<Self, super::HistErrors>
264                where S: Borrow<Self> {
265                if hists.is_empty(){
266                    return Err(HistErrors::EmptySlice);
267                }
268                let first = hists[0].borrow().binning();
269                let width = first.bin_width;
270                let mut left = first.first_border();
271                let mut right = first.last_border();
272                for other in hists[1..].iter(){
273                    let binning = other.borrow().binning();
274                    if width != binning.bin_width{
275                        return Err(HistErrors::ModuloError);
276                    }
277                    left = left.min(binning.first_border());
278                    right = right.max(binning.last_border());
279                }
280                // now I first create the binning, then I check if all the intervals aligned properly
281                let new_binning = <paste!{[<Binning $t:upper>]}>::new_inclusive(
282                    left,
283                    right,
284                    width
285                ).map_err(|_| HistErrors::ModuloError)?;
286
287                let new_first_border = to_u(new_binning.first_border());
288                let bin_width_u = first.bin_width_unsigned;
289
290                for hist in hists.iter(){
291                    let binning = hist.borrow().binning();
292                    let distance = to_u(binning.first_border()) - new_first_border;
293                    let modulo = distance % bin_width_u;
294                    if modulo != 0 {
295                        return Err(HistErrors::Alignment);
296                    }
297                }
298                Ok(GenericHist::new(new_binning))
299            }
300        }
301    };
302    (
303        $($t:ty),* $(,)?
304    ) => {
305        $(
306            other_binning!($t);
307        )*
308    }
