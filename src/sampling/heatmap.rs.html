1//! # Generate heatmaps. Create Gnuplot scripts to plot said heatmaps
2//! Note, you can use [`HeatmapUsize`] 
3//! and [`HeatmapF64`] for **all** types, for which you have
4//! implemented a histogram
5//! # Examples
6//! Here a simple Heatmap is created. 
7//! I am using random data for the heatmap, to show how it works.
8//! 
9//! **Note**: both the x and y axis will be going from 0 to 10,
10//! even though the y axis should go from 10 to 20.
11//! The reason is, that the Histograms, which the Heatmap is based upon
12//! do not have to be numbers at all. Therefore the default axis 
13//! are the bin indices.
14//! ```
15//! use {
16//!     sampling::{HistUsizeFast, HeatmapU, GnuplotSettings, GnuplotAxis, GnuplotPalette, GnuplotTerminal},
17//!     rand_pcg::Pcg64,
18//!     rand::{SeedableRng, distr::{Uniform, Distribution}},
19//!     std::{fs::File, io::BufWriter}
20//! };
21//! 
22//! // random number generator for data creation
23//! let mut rng = Pcg64::seed_from_u64(2972398345698734489);
24//! let dist_x = Uniform::new(0, 11).unwrap(); // creates random numbers between 0 and 10
25//! let dist_y = Uniform::new(10, 21).unwrap(); // creates random numbers between 10 and 20
26//! 
27//! // I am now creating two histograms, which will be 
28//! // defining the x and y axis of the heatmap
29//! let hist_x = HistUsizeFast::new_inclusive(0, 10)
30//!     .unwrap();
31//! let hist_y = HistUsizeFast::new_inclusive(10, 20)
32//!     .unwrap();
33//! // create the Heatmap
34//! let mut heat = HeatmapU::new(hist_x, hist_y);
35//! 
36//! for _ in 0..10000 {
37//!     let x = dist_x.sample(&mut rng);
38//!     let y = dist_y.sample(&mut rng);
39//!     // counting the values
40//!     heat.count(x, y).unwrap();
41//! }
42//! 
43//! // creating a file to store the gnuplot script in
44//! let heatmap_file = File::create("HeatmapU01.gp")
45//!     .expect("unable to create file");
46//! let heatmap_writer = BufWriter::new(heatmap_file);
47//! let mut settings = GnuplotSettings::default();
48//! settings.terminal(GnuplotTerminal::PDF("HeatmapU01".to_owned()));
49//! 
50//! // creating the gnuplot script
51//! heat.gnuplot(
52//!     heatmap_writer,
53//!     &settings
54//! ).unwrap();
55//! 
56//! // for correct axis, you have to set them yourself,
57//! // since the histograms do not even need to be numeric
58//! let x_axis = GnuplotAxis::new(0.0, 10.0, 6);
59//! let y_axis = GnuplotAxis::new(10.0, 20.0, 6);
60//! settings.x_axis(x_axis)
61//!     .y_axis(y_axis)
62//! // you can also change the color space, if you like
63//!     .palette(GnuplotPalette::PresetRGB)
64//!     .terminal(GnuplotTerminal::PDF("HeatmapU02".to_owned()));
65//! 
66//! // creating a file to store the gnuplot script in
67//! let heatmap_file = File::create("HeatmapU02.gp")
68//!     .expect("unable to create file");
69//! let heatmap_writer = BufWriter::new(heatmap_file);
70//!
71//! heat.gnuplot(
72//!     heatmap_writer,
73//!     settings
74//! ).unwrap();
75//! ```
76//! Now you can create the plots by calling
77//! ```bash
78//! gnuplot HeatmapU01.gp HeatmapU02.gp
79//! ```
80//! which will create `HeatmapU01.pdf` and  `HeatmapU02.pdf`
