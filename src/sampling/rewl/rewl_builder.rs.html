39/// # Errors
40/// that can arise during the construction of RewlBuilder
41#[derive(Debug)]
42pub enum RewlBuilderErr{
43    /// * The threshold for `log_f` needs to be a normal number.
44    /// * That basically means: the number is neither zero, infinite, subnormal, or NaN. 
45    ///   For more info, see the [Documentation](`std::primitive::f64::is_normal`)
46    NonNormalThreshold,
47    /// log_f_threshold must not be negative
48    Negative,
49    /// Histogram vector needs to contain at least one entry.
50    Empty,
51    /// Each histogram needs to have **at least** two bins. Though more than two bins are 
52    /// strongly recommended
53    HistBinCount,
54
55    /// Length of histogram vector and ensemble vector has to be the same!
56    LenMissmatch
57}

133    /// # new rewl builder
134    /// * used to create a **R**eplica **e**xchange **w**ang **l**andau simulation.
135    /// * use this method, if you want to have fine control over each walker, i.e., if you can
136    ///   provide ensembles, who's energy is already inside the corresponding intervals `hists`
137    /// * you might want to use [from_ensemble](crate::ReplicaExchangeWangLandauBuilder::from_ensemble) or
138    ///   [from_ensemble_tuple](crate::ReplicaExchangeWangLandauBuilder::from_ensemble_tuple) instead
139    ///
140    /// | Parameter             | meaning                                                                                                                                                  |
141    /// |-----------------------|----------------------------------------------------------------------------------------------------------------------------------------------------------|
142    /// | `ensembles`           | a vector of ensembles, one for each interval. Corresponds to the `hists` entries.                                                                        |
143    /// | `hists`               | Overlapping intervals for the wang landau walkers. Should be sorted according to their respective left bins.                                             |
144    /// | `step_size`           | step_size for the markov steps, which will be performed                                                                                                  |
145    /// | `sweep_size`          | How many steps will be performed until the replica exchanges are proposed                                                                                |
146    /// | `walker_per_interval` | How many walkers should be used for each interval (entry of `hists`)                                                                                     |
147    /// | `log_f_threshold`     | Threshold for the logarithm of the factor f (see paper). Rewl Simulation is finished, when all(!) walkers have a factor log_f smaller than this threshold |
148    ///
149    /// ## Notes
150    /// * for proper statistics, you should seed the random number generators (used for the markov chain) of all ensembles 
151    ///   differently!
152    /// * `log_f_threshold` has to be a [normal](`std::primitive::f64::is_normal`) and non negative number
153    /// * each entry of `ensembles` will be cloned `walker_per_interval - 1` times and their respective rng's will be 
154    ///   seeded via the `HasRng` trait
155    pub fn from_ensemble_vec(
156        ensembles: Vec<Ensemble>,
157        hists: Vec<Hist>,
158        step_size: usize,
159        sweep_size: NonZeroUsize,
160        walker_per_interval: NonZeroUsize,
161        log_f_threshold: f64
162    ) -> Result<Self, RewlBuilderErr>
163    {
164        if !log_f_threshold.is_normal(){
165            return Err(RewlBuilderErr::NonNormalThreshold);
166        }
167        if log_f_threshold < 0.0 {
168            return Err(RewlBuilderErr::Negative);
169        }
170        if hists.is_empty()
171        {
172            return Err(RewlBuilderErr::Empty);
173        }
174        if hists.len() != ensembles.len()
175        {
176            return Err(RewlBuilderErr::LenMissmatch);
177        }
178        if hists.iter().any(|v| v.bin_count() < 2)
179        {
180            return Err(RewlBuilderErr::HistBinCount);
181        }
182        let step_size = (0..ensembles.len())
183            .map(|_| step_size)
184            .collect();
185        
186        let mut sweep_size_vec = Vec::with_capacity(ensembles.len());
187        sweep_size_vec.extend(
188            (0..ensembles.len())
189                .map(|_| sweep_size)
190        );
191
192        let finished = vec![false; hists.len()];
193        Ok(
194            Self{
195                ensembles,
196                step_size,
197                sweep_size: sweep_size_vec,
198                walker_per_interval,
199                hists,
200                log_f_threshold,
201                s: PhantomData::<S>,
202                res: PhantomData::<Res>,
203                finished
204            }
205        )
206    }

208    /// # Create a builder to create a replica exchange wang landau (Rewl) simulation
209    /// * creates vector of ensembles and (re)seeds their respective rngs (by using the `HasRng` trait)
210    /// * calls [`Self::from_ensemble_vec(…)`](`crate::ReplicaExchangeWangLandauBuilder::from_ensemble_vec`) afterwards,
211    ///   look there for more information about the parameter
212    pub fn from_ensemble<R>(
213        ensemble: Ensemble,
214        hists: Vec<Hist>,
215        step_size: usize,
216        sweep_size: NonZeroUsize,
217        walker_per_interval: NonZeroUsize,
218        log_f_threshold: f64,
219
220    ) -> Result<Self,RewlBuilderErr>
221    where Ensemble: HasRng<R> + Clone,
222        R: Rng + SeedableRng
223    {
224        let len = NonZeroUsize::new(hists.len())
225            .ok_or(RewlBuilderErr::Empty)?;
226        let ensembles = Self::clone_and_seed_ensembles(ensemble, len);
227        Self::from_ensemble_vec(ensembles, hists, step_size, sweep_size, walker_per_interval, log_f_threshold)
228    }

230    fn clone_and_seed_ensembles<R>(mut ensemble: Ensemble, size: NonZeroUsize) -> Vec<Ensemble>
231    where Ensemble: Clone + HasRng<R>,
232        R: SeedableRng + Rng
233    {
234         let mut ensembles = (1..size.get())
235                .map(|_| {
236                    let mut e = ensemble.clone();
237                    let mut rng = R::from_rng(ensemble.rng());
238                    e.swap_rng(&mut rng);
239                    e
240                })
241                .collect::<Vec<_>>();
242        ensembles.push(ensemble);
243        ensembles
244    }

246    /// # Create a builder to create a replica exchange wang landau (Rewl) simulation
247    /// * creates vector of ensembles and (re)seeds their respective rng's (by using the `HasRng` trait).
248    ///   The vector is created by cloning `ensemble_tuple.0` for everything up to the middle of the vector and 
249    ///   `ensemble_tuple.1` for the rest. The length of the vector will be the same as `hists.len()`.
250    ///   If It is an uneven number, the middle element will be a clone of `ensemble_tuple.1`
251    /// * calls [`Self::from_ensemble_vec(…)`](`crate::ReplicaExchangeWangLandauBuilder::from_ensemble_vec`) afterwards,
252    ///   look there for more information about the parameter
253    /// * use this, if you know configurations, that would be good starting points for finding 
254    ///   configurations at either end of the intervals. 
255    pub fn from_ensemble_tuple<R>(
256        ensemble_tuple: (Ensemble, Ensemble),
257        hists: Vec<Hist>,
258        step_size: usize,
259        sweep_size: NonZeroUsize,
260        walker_per_interval: NonZeroUsize,
261        log_f_threshold: f64,
262    ) -> Result<Self,RewlBuilderErr>
263    where Ensemble: HasRng<R> + Clone,
264        R: Rng + SeedableRng
265    {
266        let len = NonZeroUsize::new(hists.len())
267            .ok_or(RewlBuilderErr::Empty)?;
268        if len < unsafe{NonZeroUsize::new_unchecked(2)} {
269            return Err(RewlBuilderErr::LenMissmatch);
270        }
271        let (left, mut right) = ensemble_tuple;
272        let mut ensembles = Vec::with_capacity(len.get());
273        let mid = len.get() / 2;
274
275        for _ in 1..mid {
276            let mut e = left.clone();
277            let mut rng = R::from_rng(right.rng());
278            e.swap_rng(&mut rng);
279            ensembles.push(e);
280        }
281        ensembles.push(left);
282        for _ in mid..len.get()-1
283        {
284            let mut e = right.clone();
285            let mut rng = R::from_rng(right.rng());
286            e.swap_rng(&mut rng);
287            ensembles.push(e);
288        }
289        ensembles.push(right);
290
291        Self::from_ensemble_vec(ensembles, hists, step_size, sweep_size, walker_per_interval, log_f_threshold)
292    }

294    #[allow(clippy::result_large_err)]
295    fn build<Energy, R, R2>
296    (
297        container: Vec<(Hist, Ensemble, Option<Energy>)>,
298        walker_per_interval: NonZeroUsize,
299        log_f_threshold: f64,
300        step_size: Vec<usize>,
301        sweep_size: Vec<NonZeroUsize>,
302        finished: Vec<bool>
303
304    ) -> Result<Rewl<Ensemble, R, Hist, Energy, S, Res>, Self>
305    where Energy: Clone,
306    R2: Rng + SeedableRng,
307    Ensemble: HasRng<R2>,
308    R: SeedableRng + Rng + Send + Sync,
309    walker::RewlWalker<R, Hist, Energy, S, Res>: Send +  Sync,
310    Hist: HistogramVal<Energy>,
311    Energy: Send + Sync
312    {
313        if container.iter().any(|(_, _, e)| e.is_none()){
314            let (hists, ensembles) = container.into_iter()
315                .map(|(h, e, _)| (h, e))
316                .unzip();
317            return Err(
318                Self{
319                    ensembles,
320                    hists,
321                    walker_per_interval,
322                    s: PhantomData::<S>,
323                    res: PhantomData::<Res>,
324                    log_f_threshold,
325                    step_size,
326                    sweep_size,
327                    finished
328                }
329            );
330        }
331
332        let mut ensembles_rw_lock = Vec::with_capacity(container.len() * walker_per_interval.get());
333        let mut walker = Vec::with_capacity(walker_per_interval.get() * container.len());
334        let mut counter = 0;
335
336        for (((mut h, mut e, energy), step_size), sweep_size) in container.into_iter()
337            .zip(step_size.into_iter())
338            .zip(sweep_size.into_iter())
339        {
340            let energy = energy.unwrap();
341            h.reset();
342            for _ in 0..walker_per_interval.get()-1 {
343                let mut ensemble = e.clone();
344                let mut rng = R2::from_rng(e.rng());
345                ensemble.swap_rng(&mut rng);
346                
347                ensembles_rw_lock.push(RwLock::new(ensemble));
348                let rng = R::from_rng(e.rng());
349                walker.push(
350                    RewlWalker::<R, Hist, Energy, S, Res>::new(
351                        counter,
352                        rng,
353                        h.clone(),
354                        sweep_size,
355                        step_size,
356                        energy.clone()
357                    )
358                );
359                counter += 1;
360            }
361            let rng = R::from_rng(e.rng());
362            walker.push(
363                RewlWalker::new(
364                    counter,
365                    rng,
366                    h,
367                    sweep_size,
368                    step_size,
369                    energy
370                )
371            );
372            counter += 1;
373            ensembles_rw_lock.push(RwLock::new(e));
374        }
375
376        let last_extreme_interval_visited = vec![ExtremeInterval::None; walker.len()];
377        let roundtrip_halves = vec![0; walker.len()];
378
379        let mut res = Rewl{
380            ensembles: ensembles_rw_lock,
381            replica_exchange_mode: true,
382            chunk_size: walker_per_interval,
383            walker,
384            log_f_threshold,
385            last_extreme_interval_visited,
386            roundtrip_halfes: roundtrip_halves
387        };
388
389        res.update_roundtrips();
390
391        Ok(
392            res
393        )
394    }

458    /// # Create `Rewl`, i.e., Replica exchange wang landau simulation
459    /// * similar to [`try_greedy_build`](`crate::ReplicaExchangeWangLandauBuilder::try_greedy_build`)
460    /// * Difference: You can choose a different `Rng` for the Wang Landau walkers (i.e., the
461    ///   acceptance of the replica exchange moves etc.)
462    /// * usage: `self.try_greedy_choose_rng_build::<RNG,_,_,_,_>(energy_fn, condition)`
463    #[allow(clippy::result_large_err)]
464    pub fn try_greedy_choose_rng_build<R, R2, F, C, Energy>(self, energy_fn: F, condition: C) -> Result<Rewl<Ensemble, R, Hist, Energy, S, Res>, Self>
465    where Hist: HistogramVal<Energy>,
466        R: Rng + SeedableRng + Send + Sync,
467        Ensemble: HasRng<R2> + Sized,
468        R2: Rng + SeedableRng,
469        F: Fn(&mut Ensemble) -> Option::<Energy> + Copy + Send + Sync,
470        C: Fn() -> bool + Copy + Send + Sync,
471        Energy: Sync + Send + Clone,
472        walker::RewlWalker<R, Hist, Energy, S, Res>: Send,
473    {
474        let mut container = Vec::with_capacity(self.hists.len());
475        let ensembles = self.ensembles;
476        let hists = self.hists;
477        let step_size = self.step_size;
478        let sweep_size = self.sweep_size;
479        let mut finished = self.finished;
480        
481        ensembles.into_par_iter()
482            .zip(hists.into_par_iter())
483            .zip(step_size.par_iter())
484            .zip(finished.par_iter_mut())
485            .zip(sweep_size.par_iter())
486            .map(
487                |((((mut e, h), &step_size), finished), sweep_size)|
488                {
489                    let mut energy = 'outer: loop
490                    {
491                        for _ in 0..sweep_size.get(){
492                            if let Some(energy) = energy_fn(&mut e){
493                                break 'outer energy;
494                            }
495                            e.m_steps_quiet(step_size);
496                        }
497                        if !condition(){
498                            return (h, e, None);
499                        }
500                    };
501
502                    if !h.is_inside(&energy) {
503                        let mut distance = h.distance(&energy);
504
505                        let mut steps = Vec::with_capacity(step_size);
506                        
507                        'outer2: loop 
508                        {
509                            for _ in 0..sweep_size.get()
510                            {
511                                e.m_steps(step_size, &mut steps);
512                                let current_energy = if let Some(energy) = energy_fn(&mut e)
513                                {
514                                    energy
515                                } else {
516                                    e.steps_rejected(&steps);
517                                    e.undo_steps_quiet(&steps);
518                                    continue;
519                                };
520    
521                                let new_distance = h.distance(&current_energy);
522                                if new_distance <= distance {
523                                    e.steps_accepted(&steps);
524                                    energy = current_energy;
525                                    distance = new_distance;
526                                    if distance == 0.0 {
527                                        break 'outer2;
528                                    }
529                                }else {
530                                    e.steps_rejected(&steps);
531                                    e.undo_steps_quiet(&steps);
532                                }
533                            }
534                            if !condition()
535                            {
536                                return (h, e, None);
537                            }
538                        }
539                    }
540                    *finished = true;
541                    (h, e, Some(energy))
542                }
543            ).collect_into_vec(&mut container);
544
545        Self::build(
546            container,
547            self.walker_per_interval,
548            self.log_f_threshold,
549            step_size,
550            sweep_size,
551            finished
552        )
553    }

557    ///   are within the required intervals, i.e., histograms
558    /// * Uses overlapping intervals. Accepts a step, if the resulting ensemble is in the same interval as before,
559    ///   or it is in an interval closer to the target interval. 
560    ///   Take a look at the [`HistogramIntervalDistance` trait](`crate::HistogramIntervalDistance`)
561    /// * `overlap` should smaller than the number of bins in your histogram. E.g. `overlap = 3` if you have 200 bins
562    /// 
563    /// ## Note
564    /// * Depending on how complex your energy landscape is, this can take a very long time,
565    ///   maybe not even terminating at all.
566    /// * You can use [`try_interval_heuristik_build`](`crate::ReplicaExchangeWangLandauBuilder::try_interval_heuristik_build`) to limit the time of the search
567    pub fn interval_heuristik_build<R, R2, F, Energy>
568    (
569        self,
570        energy_fn: F,
571        overlap: NonZeroUsize
572    ) -> Rewl<Ensemble, R, Hist, Energy, S, Res>
573    where Hist: HistogramVal<Energy> + HistogramIntervalDistance<Energy>,
574        R: Rng + SeedableRng + Send + Sync,
575        Ensemble: HasRng<R> + Sized,
576        F: Fn(&mut Ensemble) -> Option::<Energy> + Copy + Send + Sync,
577        Energy: Sync + Send + Clone,
578        walker::RewlWalker<R, Hist, Energy, S, Res>: Send,
579    {
580        match Self::try_interval_heuristik_build(self, energy_fn, || true, overlap){
581            Ok(result) => result,
582            _ => unreachable!()
583        }
584    }

637    /// # Create `Rewl`, i.e., Replica exchange wang landau simulation
638    /// * similar to [`interval_heuristik_choose_rng_build`](`crate::ReplicaExchangeWangLandauBuilder::interval_heuristik_choose_rng_build`)
639    /// * Difference: You can choose the Random number generator used for the Rewl Walkers, i.e., for 
640    ///   accepting or rejecting the markov steps and replica exchanges. 
641    /// * usage: `self.try_interval_heuristik_choose_rng_build<RNG, _,_,_,_>(energy_fn, condition, overlap)]
642    #[allow(clippy::result_large_err)]
643    pub fn try_interval_heuristik_choose_rng_build<R, R2, F, C, Energy>
644    (
645        self,
646        energy_fn: F,
647        condition: C,
648        overlap: NonZeroUsize
649    ) -> Result<Rewl<Ensemble, R, Hist, Energy, S, Res>, Self>
650    where Hist: HistogramVal<Energy> + HistogramIntervalDistance<Energy>,
651        R: Rng + SeedableRng + Send + Sync,
652        Ensemble: HasRng<R2> + Sized,
653        R2: Rng + SeedableRng,
654        F: Fn(&mut Ensemble) -> Option::<Energy> + Copy + Send + Sync,
655        C: Fn() -> bool + Copy + Send + Sync,
656        Energy: Sync + Send + Clone,
657        walker::RewlWalker<R, Hist, Energy, S, Res>: Send,
658    {
659        let mut container = Vec::with_capacity(self.hists.len());
660        let ensembles = self.ensembles;
661        let hists = self.hists;
662        let step_size = self.step_size;
663        let sweep_size = self.sweep_size;
664        let mut finished = self.finished;
665        ensembles.into_par_iter()
666            .zip(hists.into_par_iter())
667            .zip(step_size.par_iter())
668            .zip(finished.par_iter_mut())
669            .zip(sweep_size.par_iter())
670            .map(
671                |((((mut e, h), &step_size), finished), sweep_size)|
672                {
673                    let mut energy = 'outer: loop
674                    {
675                        for _ in 0..sweep_size.get(){
676                            if let Some(energy) = energy_fn(&mut e){
677                                break 'outer energy;
678                            }
679                            e.m_steps_quiet(step_size);
680                        }
681                        if !condition(){
682                            return (h, e, None);
683                        }
684                    };
685
686                    if !h.is_inside(&energy) {
687                        let mut distance = h.interval_distance_overlap(&energy, overlap);
688
689                        let mut steps = Vec::with_capacity(step_size);
690                        'outer2: loop 
691                        {
692                            for _ in 0..sweep_size.get()
693                            {
694                                e.m_steps(step_size, &mut steps);
695                                let current_energy = if let Some(energy) = energy_fn(&mut e)
696                                {
697                                    energy
698                                } else {
699                                    e.undo_steps_quiet(&steps);
700                                    continue;
701                                };
702    
703                                let new_distance = h.interval_distance_overlap(&current_energy, overlap);
704                                if new_distance <= distance {
705                                    energy = current_energy;
706                                    distance = new_distance;
707                                    if distance == 0 {
708                                        break 'outer2;
709                                    }
710                                }else {
711                                    e.undo_steps_quiet(&steps);
712                                }
713                            }
714                            if !condition()
715                            {
716                                return (h, e, None);
717                            }
718                        }
719                    }
720                    *finished = true;
721                    (h, e, Some(energy))
722                }
723            ).collect_into_vec(&mut container);
724
725        Self::build(
726            container,
727            self.walker_per_interval,
728            self.log_f_threshold,
729            step_size,
730            sweep_size,
731            finished
732        )
733    }
